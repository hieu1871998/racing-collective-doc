import {
  FavouriteButton,
  HorseAges,
  HorseImage,
  ShareOnSocials,
  ShareSquareOutlined,
  TileAdCarousel,
} from '@components'
import { ROUTES } from '@constants'
import styled from '@emotion/styled'
import { useAutoAnimate } from '@formkit/auto-animate/react'
import { Horse, Views } from '@interfaces'
import { Button } from '@ui'
import { Col, Row, Tooltip, Typography } from 'antd'
import parse, { domToReact, Element, HTMLReactParserOptions } from 'html-react-parser'
import { kebabCase } from 'lodash'
import Link from 'next/link'

const { Title } = Typography

/**
 * Get horse links from routes
 * @param horse
 * @returns {object} href, as
 */
export const getHorseLink = (horse: Horse) => {
  return {
    href: `${ROUTES.HORSE_DETAIL}/[id]/[slug]`,
    as: `${ROUTES.HORSE_DETAIL}/${horse?.id}/${
      horse?.horseName ? kebabCase(horse?.horseName) : kebabCase(horse?.name)
    }`,
  }
}

interface HorseItemProps {
  data: Horse
  view?: Views
  refetch?: () => void
  isRefetch?: boolean
  isFavourite?: boolean
  home?: boolean
}

export const HorseItem = ({ data, view, refetch, isRefetch, isFavourite, home }: HorseItemProps) => {
  const autoAnimated = useAutoAnimate(null) as React.Ref<HTMLDivElement>
  const replaceOptions: HTMLReactParserOptions = {
    replace: (domNode: Element) => {
      if (domNode?.name) {
        return <>{domToReact(domNode?.children, replaceOptions)}</>
      }
      if (domNode?.name === 'p') {
        return <>{domToReact(domNode?.children, replaceOptions)}</>
      }
      if (domNode?.name === 'strong' || (domNode?.attribs as unknown as string) === 'style') {
        return <>{domToReact(domNode?.children, replaceOptions)}</>
      }
      if (domNode?.name === 'iframe' || domNode?.name === 'img' || domNode?.name === 'br') {
        return <></>
      }
    },
  }

  return (
    <>
      {data.horseClassifieds ? (
        <ClassifiedCardWrapper
          className={`w-full border rounded bg-transparent ${
            view === 'image-list' || view === 'list'
              ? 'relative h-[4.5rem] md:h-32 lg:h-[175px] xl:h-[13.375rem] 2xl:h-64'
              : 'xl:h-[465px] lg:h-[510px] sm:h-[545px] h-[495px]'
          }
          ${home ? '2xl:w-[366px] xl:w-[280px] lg:w-[312px] sm:w-[350px]' : ''}`}
        >
          <TileAdCarousel
            data={data?.horseClassifieds}
            view={view}
            home={home}
          />
        </ClassifiedCardWrapper>
      ) : (
        <CardWrapper
          ref={autoAnimated}
          className={`w-full bg-white rounded hover:shadow-card border ${
            view === 'grid'
              ? 'xl:h-[465px] lg:h-[510px] sm:h-[545px] h-[495px]'
              : view === 'image-list'
              ? 'lg:h-64'
              : 'h-auto'
          }
          ${(!data?.approved || data.sold) && isFavourite ? 'opacity-70' : ''}
          ${home ? '2xl:w-[366px] xl:w-[280px] lg:w-[312px] sm:w-[350px]' : ''}`}
        >
          {view !== 'list' ? (
            <HorseItemContent
              view={view}
              data={data}
              refetch={refetch}
              isRefetch={isRefetch}
              replaceOptions={replaceOptions}
            />
          ) : (
            <HorseListItemContent
              view={view}
              data={data}
              refetch={refetch}
              isRefetch={isRefetch}
              replaceOptions={replaceOptions}
            />
          )}
        </CardWrapper>
      )}
    </>
  )
}

const HorseItemContent = ({
  view,
  data,
  refetch,
  isRefetch,
  replaceOptions,
}: {
  view: Views
  data: Horse
  refetch: () => void
  isRefetch?: boolean
  replaceOptions: HTMLReactParserOptions
}) => {
  return (
    <Row justify='end'>
      <Col
        className={`relative
                  ${view === 'grid' ? 'xl:h-52 lg:h-64 sm:h-72 h-60' : 'h-64 rounded-l'}`}
        span={view === 'grid' ? 24 : view === 'image-list' ? 6 : 0}
        xl={view === 'grid' ? 24 : view === 'image-list' ? 6 : 0}
        lg={view === 'grid' ? 24 : view === 'image-list' ? 8 : 0}
        xs={view === 'grid' ? 24 : view === 'image-list' ? 24 : 0}
        sm={view === 'grid' ? 24 : view === 'image-list' ? 10 : 0}
      >
        <Link
          href={getHorseLink(data).href}
          as={getHorseLink(data).as}
        >
          <a
            className={`h-full overflow-hidden block relative
                      ${!data?.approved ? 'pointer-events-none' : ''}
                      ${view === 'grid' ? 'rounded-t' : 'rounded-l'}`}
          >
            <HorseImage data={data} />
          </a>
        </Link>
        <FavouriteButton
          className='absolute top-2 right-2'
          horseId={data?.id}
          isFavourite={data?.favorite}
          refetch={refetch}
          isRefetch={isRefetch}
        />
        <div
          className={`w-full absolute bottom-0 text-right p-2 text-sm text-white
                    ${
                      data?.countryLocation || data?.state || data?.location
                        ? 'bg-gradient-to-t from-black/50 to-transparent'
                        : 'bg-none'
                    }`}
        >
          <span className='text-white'>
            <span>{data?.countryLocation ? data?.countryLocation?.name : ''}</span>
            {data?.state || data?.location ? ' | ' : ''}

            <span>{data?.location}</span>
          </span>
        </div>

      <Col

            <Title

            </Title>
          </Tooltip>
          {view === 'list' && (
            <FavouriteButton
              horseId={data?.id}
              isFavourite={data?.favorite}
              refetch={refetch}
              isRefetch={isRefetch}
            />
          )}
        </div>
        <Row>
          <Col
            lg={view === 'grid' ? 24 : 12}
            xs={view === 'grid' ? 24 : 24}
            sm={view === 'grid' ? 24 : 24}
            className={`text-xs text-neutral-400

          >
            {data?.age && <HorseAges age={data?.age} />}
            <span>{data?.colour?.colour ? ' | ' + data?.colour?.colour : ''}</span>

            </div>
