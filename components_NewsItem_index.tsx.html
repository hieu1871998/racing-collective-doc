/**@format */
import { EyeOutlined } from '@ant-design/icons'
import { ShareOnSocials, ShareSquareOutlined, TileAdCarousel } from '@components'
import styled from '@emotion/styled'
import { useAutoAnimate } from '@formkit/auto-animate/react'
import { News, Views } from '@interfaces'
import { Button, Ellipsis } from '@ui'
import { slugGen } from '@utils'
import { Col, Grid, Row, Tooltip } from 'antd'
import { Breakpoint } from 'antd/lib/_util/responsiveObserve'
import dayjs from 'dayjs'
import parse, { domToReact, Element, HTMLReactParserOptions } from 'html-react-parser'
import Image from 'next/future/image'
import Link from 'next/link'
import { NextRouter, useRouter } from 'next/router'
import React from 'react'
import defaultImage from '../../../public/assets/images/logos/thumbnail-primary.svg'

const { useBreakpoint } = Grid

/**
 * NewsItem component properties.
 * @alias NewsItemProps
 * @memberof NewsItem
 */
interface NewsItemProps {
  /**News details. */
  news: News
  /**Current view layout of the page, determine how the component is rendered. */
  view?: Views
  /**Determine whether the item is for news. */
  isNews?: true
  setPreviewVisible: React.Dispatch<React.SetStateAction<boolean>>
  /**Determine whether the item is show on homepage. */
  home?: boolean
}

/**
 * News in a type of card component, displaying headline, source, description, etc.
 * @alias NewsItem
 * @component
 * @param {NewsItemProps} NewsItemProps - `NewsItem` component properties.
 */
export const NewsItem = ({ news, view, isNews, setPreviewVisible, home }: NewsItemProps) => {
  const router = useRouter()
  const screens = useBreakpoint()
  const autoAnimated = useAutoAnimate(null) as React.Ref<HTMLDivElement>

  const replaceOptions: HTMLReactParserOptions = {
    replace: ({ name, attribs, children }: Element) => {
      if (name) {
        return <>{domToReact(children, replaceOptions)}</>
      }
      if (name === 'p') {
        return <>{domToReact(children, replaceOptions)}</>
      }
      if (name === 'strong' || (attribs as unknown as string) === 'style') {
        return <>{domToReact(children, replaceOptions)}</>
      }
      if (name === 'iframe' || name === 'img' || name === 'br') {
        return <></>
      }
    },
  }

  const onClick = (event: React.MouseEvent<HTMLDivElement | HTMLAnchorElement, MouseEvent>) => {
    event.stopPropagation()
    router.push(
      {
        pathname: router.pathname !== '/news' ? router.pathname : '/news',
        query: {
          article: news?.id,
        },
      },
      null,
      {
        shallow: true,
      }
    )
    setPreviewVisible(true)
  }

  return (
    <>
      {news.horseClassifieds ? (
        <ClassifiedCardWrapper
          className={`w-full rounded bg-white border
            ${
              view === 'image-list' || view === 'list'
                ? 'relative h-[4.5rem] md:h-32 lg:h-[175px] xl:h-[13.375rem] 2xl:h-64'
                : 'xl:h-[465px] lg:h-[510px] sm:h-[545px] h-[495px]'
            }
            ${home ? '2xl:w-[366px] xl:w-[280px] lg:w-[312px] sm:w-[350px]' : ''}`}
        >
          <TileAdCarousel
            data={news?.horseClassifieds}
            view={view}
            home={home}
          />
        </ClassifiedCardWrapper>
      ) : (
        <NewsWrapper
          ref={autoAnimated}
          className={`w-full rounded bg-white border overflow-hidden hover:shadow-card ${
            isNews
              ? 'h-[23rem]'
              : view === 'grid'
              ? 'xl:h-[465px] lg:h-[510px] sm:h-[545px] h-[495px]'
              : view === 'image-list'
              ? 'lg:h-64'
              : 'h-auto'
          }
          ${home ? '2xl:w-[366px] xl:w-[280px] lg:w-[312px] sm:w-[350px]' : ''}`}
        >
          {view !== 'list' ? (
            <NewsItemContent
              isNews={isNews}
              news={news}
              view={view}
              router={router}
              replaceOptions={replaceOptions}
              onClick={onClick}
            />
          ) : (
            <NewsListItemContent
              news={news}
              router={router}
              screens={screens}
              onClick={onClick}
              replaceOptions={replaceOptions}
            />
          )}
        </NewsWrapper>
      )}
    </>
  )
}

const NewsItemContent = ({
  isNews,
  news,
  view,
  router,
  replaceOptions,
  onClick,
}: {
  isNews?: boolean
  news: News
  view: Views
  router: NextRouter
  replaceOptions: HTMLReactParserOptions
  onClick?: (event: React.MouseEvent<HTMLDivElement | HTMLAnchorElement, MouseEvent>) => void
}) => (
  <Row justify='end'>
    <Col
      className={`relative overflow-hidden ${
        isNews ? 'h-48' : view === 'grid' ? 'xl:h-52 lg:h-64 sm:h-72 h-60 news-thumbnail' : 'h-64'
      }`}
      span={view === 'grid' ? 24 : view === 'image-list' ? 8 : 0}
      xl={view === 'grid' ? 24 : view === 'image-list' ? 6 : 0}
      lg={view === 'grid' ? 24 : view === 'image-list' ? 8 : 0}
      sm={view === 'grid' ? 24 : view === 'image-list' ? 10 : 0}
      xs={view === 'grid' ? 24 : view === 'image-list' ? 24 : 0}
    >
      <Link href={news?.newsUrl ? news.newsUrl : `${router.basePath}/news/${news?.id}/${slugGen(news?.title)}`}>
        <a
          className='block w-full h-full relative'
          target='_blank'
          rel='noreferrer'
        >
          <Image
            className='thumbnail w-full h-full relative object-center object-cover'
            // eslint-disable-next-line @typescript-eslint/no-unsafe-assignment
            src={news?.imageUrl ?? defaultImage}
            alt={news?.title ?? ''}
            priority
            fill={news?.imageUrl !== undefined}
            sizes='(max-width: 576px) 100vw,
            (max-width: 1024px) 50vw,
            25vw'
          />
        </a>
      </Link>
    </Col>
    <Col
      className={`flex flex-col ${view === 'grid' ? 'p-4' : 'px-4 py-3 lg:p-4'}`}
      span={view === 'grid' ? 24 : view === 'image-list' ? 16 : 24}
      xl={view === 'grid' ? 24 : view === 'image-list' ? 18 : 0}
      lg={view === 'grid' ? 24 : view === 'image-list' ? 16 : 0}
      sm={view === 'grid' ? 24 : view === 'image-list' ? 14 : 0}
      xs={view === 'grid' ? 24 : view === 'image-list' ? 24 : 0}
    >
      <div className='flex justify-between'>
        <Link href={news?.newsUrl ? news.newsUrl : `/news/${news?.id}/${slugGen(news?.title)}`}>
          <a
            target='_blank'
            rel='noreferrer'
          >
            <Ellipsis
              title={news?.title}
              className={`news-title cursor-pointer font-semibold text-base
                hover:text-secondary-500 transition-colors duration-300 ease-in-out ${
                  view === 'image-list' ? 'lg:mb-0 mb-2 lg:h-[] min-h-[52px]' : 'mb-2 h-[48px] leading-6'
                }`}
              maxLines={2}
            >
              {news?.title}
            </Ellipsis>
          </a>
        </Link>
      </div>
      <div className='w-full'>
        <div>
          <span
            className={`font-semibold text-sm min-h-[21px] text-primary-500 uppercase
      inline-block line-clamp-1 w-full ${view === 'image-list' ? 'md:h-auto lg:h-auto' : ''}`}
          >
            {news?.source?.name}
          </span>
          <div className='mt-2'>
            <Row className={view === 'grid' ? '' : 'h-[80px] max-h-[80px]'}>
              <p
                className={`text-sm text-neutral-400 min-h-[80px]
                  ${view === 'grid' ? 'line-clamp-4' : 'line-clamp-4'}`}
              >
                {parse(news?.description || '', replaceOptions)}
              </p>
            </Row>
            <Row>
              <Button
                className='px-0'
                type='link'
                size='small'
                onClick={onClick}
              >
                Preview
              </Button>
            </Row>
          </div>
        </div>
      </div>
      <div className={`w-full ${view === 'image-list' ? 'mt-3' : 'mt-2'}`}>
        <div className='border-t pt-2 flex items-center justify-between'>
          <div className='truncate mr-1 lg:w-[85%]'>
            {news?.publishDate && (
              <span className='news-date text-sm font-normal uppercase'>
                {dayjs(news?.publishDate).format('DD MMM YYYY')}
              </span>
            )}
          </div>
          <Tooltip
            placement='top'
            color='white'
            title={
              <ShareOnSocials
                data={news}
                shareUrl={
                  news?.newsUrl
                    ? `${process.env.NEXT_PUBLIC_ROOT_PATH}/news?article=${news?.id}`
                    : `${process.env.NEXT_PUBLIC_ROOT_PATH}/news/${news.id}/${slugGen(news?.title)}`
                }
                title={news?.title}
              />
            }
          >
            <a>
              <div className='w-6 h-6 flex justify-center items-center text-lg hover:text-secondary-500'>
                <ShareSquareOutlined />
              </div>
            </a>
          </Tooltip>
        </div>
      </div>
    </Col>
  </Row>
)
