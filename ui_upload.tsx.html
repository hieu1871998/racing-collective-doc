/**
 * Extended `Upload` component properties.
 * @alias CustomizedUploadProps
 * @memberof Upload
 */
interface CustomizedUploadProps extends UploadProps {
  /**Children node. */
  children?: React.ReactNode
  /**Set element shape.  */
  shape?: Shapes
  /**Set element layout(orientation). */
  layout?: 'horizontal' | 'vertical'
  /**Set element direction. */
  direction?: 'ltr' | 'rtl'
  /**Set border for list. */
  listTextBordered?: boolean
}

const UploadButton = ({ isLoading }: { isLoading?: boolean }) => {
  const autoAnimated = useAutoAnimate(null) as React.Ref<HTMLDivElement>

  return (
    <div ref={autoAnimated}>
      {isLoading ? <LoadingOutlined /> : <PlusOutlined />}
      <div className='m-2'>Upload</div>
    </div>
  )
}

const AvatarUploadButton = ({ isLoading }: { isLoading?: boolean }) => {
  const autoAnimated = useAutoAnimate(null) as React.Ref<HTMLDivElement>
  const [plusVisible, setPlusVisible] = useState(false)

  return (
    <div
      className='flex flex-col w-full h-full justify-center items-center'
      ref={autoAnimated}
      onMouseEnter={() => setPlusVisible(true)}
      onMouseLeave={() => setPlusVisible(false)}
    >
      <div className='text-5xl'>{isLoading ? <LoadingOutlined /> : <UserOutlined />}</div>
      {plusVisible && (
        <div className='flex text-lg gap-1 line-clamp-2'>
          <CameraOutlined />
          <PlusOutlined />
        </div>
      )}
    </div>
  )
}

const DraggerUpload = (props: CustomizedUploadProps) => {
  return <Dragger />
}

/**
 * `Ant Design` `Upload` component with custom style and request.
 * @alias Upload
 * @description
 * Upload file by selecting or dragging.
 *
 * **When To Use**
 *
 * Uploading is the process of publishing information (web pages, text, pictures, video, etc.)
 * to a remote server via a web page or upload tool.
 *  - When you need to upload one or more files.
 *  - When you need to show the process of uploading.
 *  - When you need to upload files by dragging and dropping.
 * @param {CustomizedUploadProps} props - Upload component properties.
 * See [`Upload API`](https://ant.design/components/upload/#API)
 * @component
 * @category @ui
 */
export const Upload = (props: CustomizedUploadProps) => {
  return (
    <StyledUpload
      shape={props?.shape}
      layout={props?.layout}
      direction={props?.direction}
      showUploadList={props?.showUploadList ? props?.showUploadList : ({
        previewIcon: file => {
          return (file.type.includes('video') ||
            file.type.includes('audio'))
              ? <PlayCircleFilled />
              : <EyeFilled />
        },
        removeIcon: <DeleteFilled />
      })}
      iconRender={(file, listType) => {
        if (file.type.includes('image'))
          return (
            <FileImageTwoTone
              className={listType === 'picture-card' ? 'mt-6' : ''}
              twoToneColor='#147AC4'
            />
          )
        if (file.type.includes('audio'))
          return (
            <AudioTwoTone
              className={listType === 'picture-card' ? 'mt-6' : ''}
              twoToneColor='#147AC4'
            />
          )
        if (file.type.includes('video'))
          return (
            <PlayCircleTwoTone
              className={listType === 'picture-card' ? 'mt-6' : ''}
              twoToneColor='#147AC4'
            />
          )
        if (file.type.includes('application') || file.type.includes('document'))
          return (
            <FileTextTwoTone
              className={listType === 'picture-card' ? 'mt-6' : ''}
              twoToneColor='#147AC4'
            />
          )
      }}
      {...props}
    />
  )
}
