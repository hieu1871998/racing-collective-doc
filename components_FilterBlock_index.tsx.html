components/FilterBlock/index.tsx

import { DeleteFilled, LoadingOutlined } from '@ant-design/icons'
import {
  useDeleteFilters,
  useGetSavedFilters,
  useSaveFilters
} from '@api/queries'
import { getSavedFiltersById } from '@api/services'
import { StorageKeys } from '@api/storage'
import { FieldDropdown, SignupLoginModal } from '@components'
import { Messages } from '@constants'
import { filterType } from '@constants/enum'
import { AppContext, UserContext } from '@contexts'
import styled from '@emotion/styled'
import { useAutoAnimate } from '@formkit/auto-animate/react'
import {
  DirectoryRequest,
  ErrorResponse,
  Filters,
  HorseRequest,
  LocalStorageFilters,
  NewsRequest,
  SearchSelectField
} from '@interfaces'
import {
  Button, DatePicker, Form, Input, Modal, notification, Select
} from '@ui'
import { Col, FormInstance, Grid, Row, Space, Spin } from 'antd'
import dayjs, { Dayjs } from 'dayjs'
import { find } from 'lodash'
import { useRouter } from 'next/router'
import React, {
  Dispatch,
  SetStateAction,
  useContext,
  useEffect,
  useMemo,
  useRef,
  useState
} from 'react'
import { useLocalStorage } from 'react-use'

const { useBreakpoint } = Grid

interface FilterBlockProps {
  form: FormInstance
	searchFields: SearchSelectField[]
	setCountryId?: Dispatch<SetStateAction<number>>
	setFieldFilter?: Dispatch<
		SetStateAction<{
			field: SearchSelectField
			value: string
		}>
	>
	hasSaveFilter: boolean
	onSearch?: (data: HorseRequest | NewsRequest | DirectoryRequest, showDrawer?: boolean) => void
	contentType?: string
	type?: number
	byTopic?: number[]
	setByTopic?: (id: number[]) => void,
	myRef?: React.RefObject<HTMLDivElement>
}

/**
 * Filter blocks component, render filter blocks by passing search fields
 * @component
 * @returns {JSX.Element} <FilterBlock {...props} />
 */
export const FilterBlock: React.FunctionComponent<FilterBlockProps> = ({
  form,
	searchFields,
	hasSaveFilter,
	onSearch,
	setFieldFilter,
	contentType,
	setCountryId,
	type,
	myRef
}: FilterBlockProps): JSX.Element => {
	const userContext = useContext(UserContext)
	const screens = useBreakpoint()
	const router = useRouter()
  const blockRef = useRef<HTMLDivElement>(null)
	const [date, setDate] = useState<Dayjs>()
	const [signupLoginVisible, setSignupLoginVisible] = useState(false)
	const [filtersSaveInputVisible, setFiltersSaveInputVisible] = useState(false)
	const [filtersCollectionName, setFiltersCollectionName] = useState(
		dayjs().format('DD-MM-YYYY')
	)
	const [filters, setFilters, removeFilters] =
		useLocalStorage<LocalStorageFilters>(
			contentType === 'horse'
      ? `${StorageKeys.HORSE_FILTERS}-${type}`
      : `${StorageKeys.NEWS_FILTERS}-${type}`
      ,
			{
				type: type,
        title: undefined,
				body: undefined,
			}
		)
	const [searchTitleId, setSearchTitleId] = useState<number>(
		parseInt((router.query.searchId as string) || undefined)
	)
	const animated = useAutoAnimate(null) as React.Ref<HTMLDivElement>
	const countryId = Form.useWatch<number>('countryId', form)
  	const tagContext = useContext(AppContext)
	const [savedSearchPage, setSavedSearchPage] = useState<boolean>(router.query.searchId ? true : false)
	const isLoggedIn = useMemo(() => {
		if (userContext.contextLoaded) {
			const user = userContext.currentUser
			return !!user
		}
	}, [userContext.contextLoaded, userContext.currentUser])

	const {
		data: savedFiltersResp,
		refetch: fetchSavedFilters,
		isFetching: fetchingSavedFilters,
	} = useGetSavedFilters(type, isLoggedIn)
	const savedFilters = useMemo(
		() => savedFiltersResp?.data?.responseData,
		[savedFiltersResp]
	)

	const onFinish = (data: Filters) => {
		let prices: string[] = []

		if (data?.price) {
			prices = data.price.split('-')
			delete data.price
			data.minPrice = prices[0]
			data.maxPrice = prices[1]
		}

		setFilters(values => ({
			...values,
      title: form.getFieldValue('searchTitle') as number,
			body: data
		}))
		onSearch(data, savedSearchPage)
	}

	useEffect(() => {
		!!setCountryId && setCountryId(countryId)
		if (countryId === undefined) {
			form.resetFields(['stateIds'])
		}
	}, [countryId, setCountryId, form])

	useEffect(() => {
		if (filters?.body && filters?.type === type) {
			onSearch(filters.body, true)
		}
	// eslint-disable-next-line react-hooks/exhaustive-deps
	}, [filters, type])

	useEffect(() => {
		if (filters?.body && !savedSearchPage) {
			let localPrice: string = undefined
      let localDate: Dayjs = undefined
      if (filters?.body?.minPrice || filters?.body?.maxPrice) {
        localPrice =
          (filters?.body.minPrice || '0') +
          (!!filters?.body?.maxPrice && '-') +
          (filters?.body.maxPrice || '')
      }

      if (filters?.body?.date) {
        localDate = dayjs(filters?.body?.date)
        setDate(localDate)
      }

      searchFields?.map(field => {
        if (field?.labelInValue && !!field?.fetch) {
          field?.fetch()
        }
      })

      filters?.title && fetchSavedFilters()

      savedFilters ? form.setFieldsValue({
        ...filters?.body,
        searchTitle: find(savedFilters, { id: filters?.title}) ? filters?.title : undefined,
        price: localPrice,
        date: localDate,
      }) : form.setFieldsValue({
        ...filters.body,
        price: localPrice,
        date: localDate,
      })
		}
	// eslint-disable-next-line react-hooks/exhaustive-deps
	}, [filters, form, type, fetchSavedFilters, savedFilters, contentType])

	const onClear = () => {
		removeFilters()
    tagContext.setTag(undefined)
		form.resetFields()
		onFinish(undefined)
	}

	const onSaveFiltersClick = () => {
		setFiltersCollectionName(dayjs().format('DD-MM-YYYY'))
		setFiltersSaveInputVisible(!filtersSaveInputVisible)
	}

	const onSaveFiltersCancel = () => {
		setFiltersSaveInputVisible(false)
	}

	const { mutateAsync: saveFilters } = useSaveFilters({
		onSuccess: async resp => {
			if (resp.data.responseData) {
				notification.success({
          message: Messages.SaveSearchSuccess
        })
				setFiltersSaveInputVisible(false)
				form.resetFields()
				await fetchSavedFilters()
				form.setFieldsValue({
					searchTitle: resp?.data?.responseData?.id,
					...resp?.data?.responseData?.body,
          date: dayjs(resp?.data?.responseData?.body?.date?.[0] as string | number | dayjs.Dayjs | Date)
				})
			} else if (resp.data.error) {
				notification.error({
          message: resp.data.error.message
        })
			}
		},
		onError: (error: ErrorResponse) => {
			notification.error({
				message: error.message,
			})
		},
	})

	const { mutateAsync: deleteFilters } = useDeleteFilters({
		onSuccess: resp => {
      const searchResp = resp?.data?.responseData
      const error = resp?.data?.error

			if (searchResp) {
				notification.success({
          message: Messages.DeleteSaveSearchSuccess
        })
				setSearchTitleId(undefined)
				form.resetFields()
				onSearch(form.getFieldsValue() as HorseRequest | NewsRequest | DirectoryRequest)
				fetchSavedFilters()
				removeFilters()
			} else if (error) {
				console.error(error)
				notification.error({
          message: error.message
        })
			}
		},
		onError: error => {
			console.error(error)
			notification.error({
        message: error.message
      })
		},
	})

	const onSaveFilters = async (
		event: React.MouseEvent<HTMLElement, MouseEvent>
	) => {
		event.stopPropagation()

		let prices: string[] = []

		const filtersToServer = form.getFieldsValue() as Filters

		if (filtersToServer.price) {
			prices = filtersToServer.price.split('-')
			filtersToServer.minPrice = prices[0]
			filtersToServer.maxPrice = prices[1]
		}

		if (isLoggedIn) {
			delete filtersToServer.searchTitle
			await saveFilters({
				body: {
          ...filtersToServer,
          date: filtersToServer?.date ? [dayjs(filtersToServer?.date)?.valueOf()] : undefined
        },
				type: type,
				name: filtersCollectionName,
			})

			onFinish(filtersToServer)
		} else {
			setSignupLoginVisible(true)
      notification.warn({
        message: Messages.LoginRequired
      })
		}
	}

	const onDeleteFilters = (id: number) => {
		Modal.confirm({
			type: 'warning',
			title: 'Confirm',
			content: Messages.ConfirmDeleteSaveSearch,
			okText: 'Yes',
			cancelText: 'No',
			onOk: () => {
				if (savedFilters === undefined) {
					form.resetFields()
				}
				return deleteFilters(id)
			},
		})
	}

	useEffect(() => {
		const searchId = router.query.searchId as string
		if (searchId) {
			onSavedFiltersSelectChange(parseInt(searchId))
			if(myRef?.current){
				window.scrollTo(0, myRef.current.offsetTop - 180);
			}
		}
	// eslint-disable-next-line react-hooks/exhaustive-deps
	}, [router.query.searchId])

	const onSavedFiltersSelectChange = async (id: number) => {
		form.resetFields()
		const savedResp = await getSavedFiltersById(id)
		const saved = savedResp?.data?.responseData?.body
		searchFields?.map(field => {
			if (field?.labelInValue && !!field?.fetch) {
				field?.fetch()
			}
		})

		if (saved) {
			let savedPrice: string = undefined
      let savedDate: Dayjs = undefined

			if (saved?.minPrice || saved?.maxPrice) {
				savedPrice =
					(saved.minPrice || '0') +
					(!!saved?.maxPrice && '-') +
					(saved.maxPrice || '')
			}

      if (saved?.date) {
        savedDate = dayjs(saved.date[0] as string | number | Date | dayjs.Dayjs)
      }

			setFilters(values => ({
				...values,
        title: id,
				body: {
          ...saved,
          price: savedPrice,
          date: savedDate
        },
			}))

			form.setFieldsValue({
				...saved,
				searchTitle: id,
				price: savedPrice,
        date: savedDate
			})

			onFinish({
        ...saved,
				price: savedPrice,
        date: savedDate
      })
	  setSavedSearchPage(false)
		}
	}

  useEffect(
    () => {
      const tag = tagContext.tag

      if (tag?.page === router.pathname) {
        form.resetFields()
        searchFields?.find(field => field.name === filterType[tag.filterType]).fetch()
        form.setFieldsValue({
          [filterType[tag.filterType]]: [tag.id]
        })
        blockRef.current.scrollIntoView()
        onFinish({
          [filterType[tag.filterType]]: [tag.id]
        })
      }
    },
    // eslint-disable-next-line react-hooks/exhaustive-deps
    [tagContext, form]
  )

	return (
		<FilterBlockWrapper ref={blockRef}>
			<Form
				className='mt-4'
				form={form}
        size='large'
				onFinish={onFinish}
			>
				<Row
					gutter={16}
					ref={screens.xs ? animated : undefined}
				>
					<Col
						xl={3}
						lg={4}
						sm={6}
						span={24}
					>
						<Form.Item name='searchTitle'>
							<Select
								size='large'
								placeholder='Saved Searches'
								optionLabelProp='label'
								getPopupContainer={(elem: HTMLElement) => elem}
								dropdownMatchSelectWidth={false}
								onFocus={() => fetchSavedFilters()}
								onSelect={(value: number) => {
									setSearchTitleId(value)
									onSavedFiltersSelectChange(value)
								}}
								disabled={!isLoggedIn}
								value={searchTitleId}
								loading={fetchingSavedFilters}
							>
								{savedFilters ? (
									savedFilters?.map(item => {
										return (
											<Select.Option
												value={item?.id}
												label={item?.name}
												key={item?.id}
											>
												<div className='flex flex-row justify-between'>
													<span className='truncate'>{item?.name}</span>
													<div
														onClick={event => {
															event.stopPropagation()
															onDeleteFilters(item?.id)
														}}
													>
														<a>
															<DeleteFilled />
														</a>
													</div>
												</div>
											</Select.Option>
										)
									})
								) : (
									<Select.Option>
									  <div className='flex justify-center items-center'>
  										<Spin
  											size='large'
  											indicator={<LoadingOutlined />}
  										/>
  									</div>
									</Select.Option>
								)}
							</Select>
						</Form.Item>
					</Col>
					{searchFields.map((field, index) => (
						<FieldDropdown
							key={field.name}
              filter
							field={field}
							formInstance={form}
							index={index}
							setFilters={setFilters}
							removeFilters={removeFilters}
							setFieldFilter={setFieldFilter}
						/>
					))}
					{contentType === 'news' && (
						<Col
							xl={3}
							lg={4}
							sm={6}
							span={24}
							order={searchFields.length + 1}
						>
							<Form.Item name='date'>
								<DatePicker
                  className='w-full'
									picker='month'
									placeholder='Select Month'
									format='YYYY MMMM'
									size='large'
									value={date}
									onChange={value => setDate(value)}
									allowClear={true}
                  inputReadOnly
                  getPopupContainer={trigger => trigger}
								/>
							</Form.Item>
						</Col>
					)}
          {!(router.pathname.includes('search')) && (
            <Col
              xl={6}
              lg={8}
              sm={12}
              span={24}
              order={screens.xs ? 1 : searchFields.length + 1}
            >
              <Form.Item name={contentType === 'news' ? 'keyWord' : 'query'}>
                <Input
                  placeholder='Search keyword'
                  size='large'
                />
              </Form.Item>
            </Col>
          )}
					<Col
						xl={6}
						lg={8}
						sm={12}
						span={24}
						order={screens.xs ? undefined : searchFields.length + 2}
					>
						<Form.Item>
							<Row>
								<Space size={'large'}>
									<Button
										type='primary'
										size='large'
										htmlType='submit'
									>
										Search
									</Button>
									<a
										className='underline'
										onClick={onClear}
									>
										Clear all
									</a>
									<a
										className={`underline ${
											hasSaveFilter && screens.xs ? 'block' : 'hidden'
										}`}
										onClick={onSaveFiltersClick}
									>
										Save filter
									</a>
								</Space>
							</Row>
						</Form.Item>
					</Col>
					<Col
						sm={24}
						span={0}
						order={screens.xs ? undefined : searchFields.length + 3}
					>
						<Form.Item
							className={`underline ${hasSaveFilter ? 'block' : 'hidden'}`}
						>
							<a onClick={onSaveFiltersClick}>Save filter</a>
						</Form.Item>
					</Col>
					{filtersSaveInputVisible && (
						<Col
							span={24}
							order={screens.xs ? 0 : searchFields.length + 4}
						>
							<Space className='flex justify-start items-start'>
								<Form.Item>
									<Input
										size='large'
										placeholder='Input Name'
										value={filtersCollectionName}
										onChange={event =>
											setFiltersCollectionName(event.currentTarget.value)
										}
									/>
								</Form.Item>
								<Form.Item>
									<Button
										size='large'
										type='primary'
										htmlType='button'
										disabled={
											filtersCollectionName === null ||
											filtersCollectionName === undefined ||
											filtersCollectionName.length === 0 ||
											!filtersCollectionName.replace(/\s/g, '').length
										}
										onClick={event => onSaveFilters(event)}
									>
										Save
									</Button>
								</Form.Item>
								<Form.Item>
									<Button
										size='large'
										type='default'
										onClick={onSaveFiltersCancel}
									>
										Cancel
									</Button>
								</Form.Item>
							</Space>
						</Col>
					)}
				</Row>
			</Form>
			<SignupLoginModal
				open={signupLoginVisible}
				setVisible={setSignupLoginVisible}
			/>
		</FilterBlockWrapper>
	)
}

const FilterBlockWrapper = styled.div`
	.ant-select {
		> .ant-select-selector {
			.ant-select-selection-overflow {
				flex-wrap: nowrap;
				overflow: hidden;
			}
		}
	}
`

export default FilterBlock
