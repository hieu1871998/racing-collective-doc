import createCache from '@emotion/cache'
import { CacheProvider } from '@emotion/react'
import styled from '@emotion/styled'
import { HomeBanner, HrefTarget } from '@interfaces'
import { Grid, Typography } from 'antd'
import { Breakpoint } from 'antd/lib/_util/responsiveObserve'
import Image from 'next/image'
import Link from 'next/link'
import { NextRouter, useRouter } from 'next/router'
import defaultImage from '../../../public/assets/images/background/default.jpg'

const { useBreakpoint } = Grid

const emotionCache = createCache({
  key: 'home-banner-item'
})

/**
 * Object containing URL string and href target.
 */
interface AnchorInterface {
  /**URL string. */
  href: string
  /**Href target. */
  target: HrefTarget
}

/**
 * An uitily/workaround for replacing internal URL hostname to work with any enviroment (dev/prod)
 * and determining href target (internal/external).
 * @alias getUrl
 * @memberof HomeBannerItem
 * @param link - URL string.
 * @param router - Next.js router.
 * @returns {AnchorInterface} Object containing URL string and href target.
 */
const getURL = (
  link: string,
  router?: NextRouter
): AnchorInterface => {
  const regex = new RegExp('^[^#]*?://.*?(/.*)$')

  if (link?.includes('bloodstockexchange')) {
    const newLink = `${router.basePath}${link?.match(regex)?.[1]}`

    return {
      href: newLink,
      target: '_self'
    }
  } else if (link) {
    return {
      href: link,
      target: '_blank'
    }
  } else {
    return {
      href: undefined,
      target: '_self'
    }
  }
}


/**
 * HomeBannerCarousel individual item, displaying banner image and titles and navigations
 * to both internal and external URLs.
 * @param {*} props - Component properties.
 * @returns {JSX.Element} HomeBannerItem component.
 */
export const HomeBannerItem = ({
  banner
}: {
  banner: HomeBanner
}): JSX.Element => {
  const router = useRouter()
  const screens = useBreakpoint()

  const URL = getURL(banner?.link, router)

  return (
    <CacheProvider value={emotionCache}>
      <ItemWrapper
        className='lg:h-[calc(100vw/3)] h-[calc(100vh/3)]'
        hasLink={!!banner?.link}
        screens={screens}
      >
        <Image
          src={
            banner?.media?.url ??
            banner?.media?.cdnOrigin ??
            defaultImage
          }
          alt={
            banner?.title ??
            banner?.titleLine1 ??
            banner?.titleLine2
          }
          layout='fill'
          objectPosition='center'
          objectFit='cover'
          priority
        />
        <div className='absolute top-0 right-0 left-0 h-40 bg-gradient-to-b from-black to-transparent opacity-50' />
        <Link href={banner?.link ? URL.href : ''}>
          <a
            className={`
              w-full h-full flex items-end absolute top-0 left-0 right-0
              ${banner?.link ? '' : 'pointer-events-none'}
            `}
            target={URL.target}
            rel='noreferrer'
          >
            {(banner?.titleLine1 || banner?.titleLine2) && (
              <div className='w-full bg-gradient-to-t from-black to-transparent'>
                <div className='container p-4 mx-auto'>
                  <Typography.Title
                    className={`uppercase line-clamp-1
                    2xl:max-w-screen-xl xl:max-w-screen-lg lg:max-w-screen-md md:max-w-screen-sm`}
                    level={screens.xl ? 1 : screens.sm ? 3 : 4}
                  >
                    {banner.titleLine1}
                  </Typography.Title>
                  <Typography.Text
                    className={
                      `line-clamp-2 xl:text-xl sm:text-sm xs:text-xs
                      2xl:max-w-screen-xl xl:max-w-screen-lg lg:max-w-screen-md md:max-w-screen-sm`
                    }
                    strong
                  >
                    {banner.titleLine2}
                  </Typography.Text>
                </div>
              </div>
            )}
          </a>
        </Link>
      </ItemWrapper>
    </CacheProvider>
  )
}

/**
 * Styled properties for ItemWrapper.
 * @alias ItemWrapperStyledProps
 * @memberof HomeBannerItem
 */
interface ItemWrapperStyledProps {
  /**Determine whether to show pointer cursor or normal cursor. */
  hasLink?: boolean
  /**Screen breakpoints */
  screens?: Partial<Record<Breakpoint, boolean>>
}

/**
 * A modified division wrapper for HomeBannerItem.
 * @param {ItemWrapperStyledProps} props - Properties for conditionally set CSS properties.
 */
const ItemWrapper = styled.div<ItemWrapperStyledProps>`
  width: 100%;
  /* height: ${props => props.screens.lg ? 'calc(100vw / 3)' : 'calc(100vh / 3)'}; */
  position: relative;
  
  a {
    cursor: ${props => props.hasLink ? 'pointer' : 'default'};

    .container {      
      .ant-typography {
        color: #ffffff;
        transition: all .3s cubic-bezier(0.5, 0, 0.5, 1);
      }
    }
  }
`
