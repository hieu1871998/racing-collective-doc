containers/news/newsDetail.tsx

import { useGetNewsList } from '@api'
import { NewsItem, NewsPreview, PageWrapper, ShareOnSocials } from '@components'
import NewsMedia from '@components/NewsMedia'
import styled from '@emotion/styled'
import { News, NewsRequest } from '@interfaces'
import { Carousel } from '@ui'
import { getMetaTag } from '@utils'
import { Grid, Row, Tag, Typography } from 'antd'
import { Breakpoint } from 'antd/lib/_util/responsiveObserve'
import dayjs from 'dayjs'
import parse from 'html-react-parser'
import { dropRight } from 'lodash'
import { NextSeo } from 'next-seo'
import Image from 'next/future/image'
import { useRouter } from 'next/router'
import { useMemo, useState } from 'react'
const { Paragraph } = Typography
const { useBreakpoint } = Grid

/**
 * NewsDetail properties interface.
 * @alias NewsDetailProps
 * @memberof NewsDetail
 */
interface NewsDetailProps {
  /**News data. */
  news: News
}

/**
 * Component for displaying news detail including 16 most recent news.
 * @alias NewsDetail
 * @author Vi Hieu
 * @component
 * @param {NewsDetailProps} { news } - News data.
 * @returns {JSX.Element} NewsDetail component.
 * @example
 * const news
 *
 * return (
 *  <NewsDetail news={news} />
 * )
 */
export const NewsDetail = ({ news }: NewsDetailProps): JSX.Element => {
  const router = useRouter()
  const screens = useBreakpoint()
  const [previewVisible, setPreviewVisible] = useState(false)

  const params: NewsRequest = {
    orderBy: 'publishedDate',
    orderDirection: 'desc',
    pageIndex: 0,
    pageSize: 20
  }

  const { data: newsResp } = useGetNewsList(params)

  const newsList = useMemo(
    () => {
      const list = newsResp?.pages?.[0]?.data?.responseData
      ?.filter(item => !item?.publicType)
      ?.filter(item => item?.id !== news?.id)

      return dropRight(list, list.length - 16)
    },
    [newsResp, news]
  )

  return (
    <PageWrapper>
      <NextSeo
        title={`${getMetaTag('head-title', news)} | Racing Collective`}
        description={getMetaTag('description', news)}
        openGraph={{
          type: 'website',
          url: getMetaTag('url', news),
          title: getMetaTag('title', news),
          description: getMetaTag('description', news),
          images: [
            {
              url: getMetaTag('image', news),
              width: 800,
              height: 600,
              alt: getMetaTag('title', news),
            },
          ],
        }}
        twitter={{
          handle: '@racingcollective',
          site: '@racingcollective',
          cardType: 'summary_large_image',
        }}
        canonical={getMetaTag('url', news)}
      />
      <NewsDetailWrapper
        className='w-full h-full bg-white sm:py-4 py-2'
        screens={screens}
      >
        <div className='container mx-auto p-4 bg-white'>
          {news?.imageUrl ? (
            <div className='w-full max-w-screen-lg mx-auto lg:px-0 sm:px-4 p-0'>
              <Image
                className='w-full h-auto object-center object-cover'
                src={news?.imageUrl}
                alt={news?.title}
                width={1600}
                height={1600}
                quality={100}
                priority
              />
            </div>
          ) : (
            null
          )}
          <div className='max-w-screen-md mx-auto sm:p-4 p-0'>
            <h1
              className='my-2 text-xl lg:text-2xl font-semibold leading-snug'
            >
              {news?.title}
            </h1>
            <Paragraph>
              <span
                className='text-[#6d6c6c] text-xs sm:text-sm'
              >
                {dayjs(news?.publishDate).format('dddd, DD MMM YYYY')}
              </span>
            </Paragraph>
            <div className='mb-4'>
              {news?.listCategory?.map(tag => (
                <Tag key={tag.id} color='var(--primary)'>
                  <a>{tag.name}</a>
                </Tag>
              ))}
            </div>
            <Paragraph className='text-xs sm:text-sm leading-6'>
              {parse(news?.content ?? '')}
            </Paragraph>
            {news?.attachment?.length && (
              <NewsMedia medias={news.attachment}/>
            )}

            <Row className='mt-4'>
              <ShareOnSocials
                data={news}
                shareUrl={process.env.NEXT_PUBLIC_ROOT_PATH + router.asPath}
                iconSize={screens.sm ? 36 : 28}
                spaceSize='small'
              />
            </Row>
          </div>
          <div className='border-t border-black border-opacity-30 mb-4 mx-1' />
          <Carousel
            slidesToShow={
              screens.xl ? 4 :
              screens.lg ? 3 :
              screens.md ? 2 :
              screens.sm ? 1 : 1
            }
            slidesToScroll={
              screens.xl ? 4 :
              screens.lg ? 3 :
              screens.md ? 2 :
              screens.sm ? 1 : 1
            }
            autoplay
            autoplaySpeed={5000}
            dots
          >
            {newsList?.map(item => (
              <NewsItem
                key={item?.id}
                news={item}
                view='grid'
                setPreviewVisible={setPreviewVisible}
              />
            ))}
          </Carousel>
        </div>
      </NewsDetailWrapper>
      <NewsPreview
        id={parseInt(router?.query?.article as string)}
        open={previewVisible && !!router?.query?.article}
        setVisible={setPreviewVisible}
      />
    </PageWrapper>

  )
}

/**
 * NewsDetail styled division wrapper propeties.
 */
interface NewsDetailWrapperStyleProps {
  /**
   * Screen breakpoints.
   * @see [useBreakpoint](https://ant.design/components/grid/#components-grid-demo-useBreakpoint)
   */
  screens?: Partial<Record<Breakpoint, boolean>>
}

/**
 * NewsDetail styled division wrapper.
 * @alias NewsDetailWrapper
 * @memberof NewsDetail
 */
const NewsDetailWrapper = styled.div<NewsDetailWrapperStyleProps>`
  .ant-typography {
    figure {
      margin: 32px 0;
      width: 100%;
      
      iframe {
        width: 100%;
        max-width: 100%;
        height: ${props => props.screens.lg ? 'calc(992px / 16 * 9)' : 'calc(100vw / 16 * 9)'};
      }
    }
  }

  .shadow-lg {
    box-shadow: 0 1px 2px -2px rgb(0 0 0 / 16%), 0 3px 6px 0 rgb(0 0 0 / 12%), 0 5px 12px 4px rgb(0 0 0 / 9%);
  }

  .ant-carousel {
    .slick-slider {
      .slick-list {
        .slick-track {
          div {
            &.slick-slide {
              padding: 8px;
            }
          }
        }
      }
    }

    .slick-dots {
      bottom: ${props => props.screens.xs ? '-14px' : '-18px'};

      li {
        width: ${props => props.screens.xs ? '8px' : '12px'};
        height: ${props => props.screens.xs ? '8px' : '12px'};
        cursor: pointer;

        button {
          width: ${props => props.screens.xs ? '8px' : '12px'};
          height: ${props => props.screens.xs ? '8px' : '12px'};
          border-radius: 33px;
          background-color: var(--primary);
        }
      }
    }
  }
`

export default NewsDetail
