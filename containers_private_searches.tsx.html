containers/private/searches.tsx

import {
  AppstoreOutlined,
  ExclamationCircleOutlined,
  LoadingOutlined
} from '@ant-design/icons'
import {
  useDeleteFilters,
  useEditFilters,
  useGetSavedFilters,
  useGetSavedFiltersById,
  useSaveFilters
} from '@api'
import { SearchesForm } from '@components'
import { Messages } from '@constants'
import styled from '@emotion/styled'
import { useAutoAnimate } from '@formkit/auto-animate/react'
import { ErrorResponse, HorseFilters, NewsRequest } from '@interfaces'
import { Form, Menu, Modal, notification, Spin } from '@ui'
import { Col, MenuProps, Row } from 'antd'
import dayjs from 'dayjs'
import { useRouter } from 'next/router'
import React, { useEffect, useMemo, useState } from 'react'

type MenuItem = Required<MenuProps>['items'][number]

enum DataType {
  'sale' = 1,
  'lease' = 2,
  'syndication' = 3,
  'news' = 8
}

enum Pages {
  Sale = 'sale',
  Lease = 'lease',
  Syndication = 'syndication',
  News = 'news'
}

export const SearchesTab = () => {
  const [form] = Form.useForm()
  const [openKeys, setOpenKeys] = useState([Pages.Sale as string])
  const [type, setType] = useState(DataType.sale)
  const [page, setPage] = useState<Pages>(Pages.Sale)
  const router = useRouter()
  const autoAnimated = useAutoAnimate(null) as React.Ref<HTMLDivElement>

  /**Open keys handling */
  const onOpenChange: MenuProps['onOpenChange'] = keys => {
    const currentKey = keys.find(key => openKeys.indexOf(key) === -1)

    if (rootSubmenuKeys.indexOf(currentKey) === -1) {
      setOpenKeys(keys)
    } else {
      setOpenKeys(currentKey ? [currentKey] : [])
    }

    /**Set first item id of searches collection as selected key.
     * Moved from useEffect as it automatically did this whenever the collection changes.
     */
    switch (currentKey) {
      case Pages.Sale:
        setType(DataType.sale)
        setPage(Pages.Sale)
        fetchSaleSearches()

        if (saleSearches?.length > 0) {
          setSelectedKeys([saleSearches?.[0]?.id?.toString()])
        } else {
          setSelectedKeys(undefined)
          form.resetFields()
        }

        break
      case Pages.Lease:
        setType(DataType.lease)
        setPage(Pages.Lease)
        fetchLeaseSearches()

        if (leaseSearches?.length > 0) {
          setSelectedKeys([leaseSearches?.[0]?.id?.toString()])
        } else {
          setSelectedKeys(undefined)
          form.resetFields()
        }

        break
      case Pages.Syndication:
        setType(DataType.syndication)
        setPage(Pages.Syndication)
        fetchSyndicationSearches()

        if (syndicationSearches?.length > 0) {
          setSelectedKeys([syndicationSearches?.[0]?.id?.toString()])
        } else {
          setSelectedKeys(undefined)
          form.resetFields()
        }

        break
      case Pages.News:
        setType(DataType.news)
        setPage(Pages.News)
        fetchNewsSearches()

        if (newsSearches?.length > 0) {
          setSelectedKeys([newsSearches?.[0]?.id?.toString()])
        } else {
          setSelectedKeys(undefined)
          form.resetFields()
        }

        break
      default:
        setType(undefined)
        setPage(undefined)
        setSelectedKeys(undefined)
    }
    form.resetFields()
  }

  /**Selected keys handling */
  const [selectedKeys, setSelectedKeys] = useState<string[]>()
  const onSelectedChange: MenuProps['onSelect'] = selectInfo => {
    form.resetFields()
    setSelectedKeys([selectInfo.key])
  }

  /**Searches collections queries */
  const {
    data: saleSearchesResp,
    refetch: fetchSaleSearches,
    isFetching: fetchingSaleSearches
  } = useGetSavedFilters(DataType.sale)
  const saleSearches = useMemo(
    () => saleSearchesResp?.data?.responseData || [],
    [saleSearchesResp]
  )

  const {
    data: leaseSearchesResp,
    refetch: fetchLeaseSearches,
    isFetching: fetchingLeaseSearches
  } = useGetSavedFilters(DataType.lease)
  const leaseSearches = useMemo(
    () => leaseSearchesResp?.data?.responseData || [],
    [leaseSearchesResp]
  )

  const {
    data: syndicationSearchesResp,
    refetch: fetchSyndicationSearches,
    isFetching: fetchingSyndicationSearches
  } = useGetSavedFilters(DataType.syndication)
  const syndicationSearches = useMemo(
    () => syndicationSearchesResp?.data?.responseData || [],
    [syndicationSearchesResp]
  )

  const {
    data: newsSearchesResp,
    refetch: fetchNewsSearches,
    isFetching: fetchingNewsSearches
  } = useGetSavedFilters(DataType.news)
  const newsSearches = useMemo(
    () => newsSearchesResp?.data?.responseData || [],
    [newsSearchesResp]
  )

  useEffect(
    () => {
      if (!selectedKeys && saleSearches?.length > 0 && openKeys?.[0] === 'sale') {
        setSelectedKeys([saleSearches?.[0]?.id?.toString()])
      }
    },
    [saleSearches, selectedKeys, openKeys]
  )

  /**Searches collection queries by ids */
  const {
    data: searchesResp,
    isFetching: fetchingSearches
  } = useGetSavedFiltersById(parseInt(selectedKeys?.[0]))
  const searches = useMemo(
    () => searchesResp?.data?.responseData,
    [searchesResp]
  )

  /**Apply searches to form */
  useEffect(
    () => {
      if (searches && !fetchingSearches) {
        if (searches?.body?.date) {
          searches.body.date = dayjs(searches.body.date[0] as string | number | Date | dayjs.Dayjs)
        }

        if (searches?.body?.minPrice || searches?.body?.maxPrice) {
          searches.body.price =
          `${searches?.body?.minPrice ?? '0'}${searches?.body?.maxPrice ? '-' : ''}${searches?.body?.maxPrice ?? ''}`
        }

        form.setFieldsValue({
          name: searches?.name,
          ...searches?.body
        })
      }
    },
    [form, searches, selectedKeys, fetchingSearches]
  )

  /**Get menu and submenu items */
  const getMenuItem = (
    label: React.ReactNode,
    key: React.Key,
    icon?: React.ReactNode,
    children?: MenuItem[],
    type?: 'group',
    disabled?: boolean
  ) => {
    return {
      label,
      key,
      icon,
      children,
      type,
      disabled
    } as MenuItem
  }

  const items: MenuProps['items'] = useMemo(
    () => ([
      getMenuItem(
        <span className='uppercase'>{Pages.Sale}</span>,
        Pages.Sale,
        <AppstoreOutlined />,
        (saleSearches?.length > 0 && openKeys[0] == Pages.Sale)
        ? saleSearches?.map(search => (
          getMenuItem(search?.name, search?.id)
        ))
        : fetchingSaleSearches
        ? [getMenuItem(<Spin indicator={<LoadingOutlined />} />, null)]
        : [getMenuItem('No saved searches', `${Pages.Sale}${type}`, null, null, null, true)],
      ),
      getMenuItem(
        <span className='uppercase'>{Pages.Lease}</span>,
        Pages.Lease,
        <AppstoreOutlined />,
        (openKeys[0] == Pages.Lease && leaseSearches?.length > 0)
        ? leaseSearches?.map(search => (
          getMenuItem(search?.name, search?.id)
        ))
        : fetchingLeaseSearches
        ? [getMenuItem(<Spin indicator={<LoadingOutlined />} />, null)]
        : [getMenuItem('No saved searches', `${Pages.Lease}${type}`, null, null, null, true)],
      ),
      getMenuItem(
        <span className='uppercase'>{Pages.Syndication}</span>,
        Pages.Syndication,
        <AppstoreOutlined />,
        (openKeys[0] == Pages.Syndication && syndicationSearches?.length > 0)
        ? syndicationSearches?.map(search => (
          getMenuItem(search?.name, search?.id)
        ))
        : fetchingSyndicationSearches
        ? [getMenuItem(<Spin indicator={<LoadingOutlined />} />, null)]
        : [getMenuItem('No saved searches', `${Pages.Syndication}${type}`, null, null, null, true)],
      ),
      getMenuItem(
        <span className='uppercase'>{Pages.News}</span>,
        Pages.News,
        <AppstoreOutlined />,
        (openKeys[0] == Pages.News && newsSearches?.length > 0)
        ? newsSearches.map(search => (
          getMenuItem(search.name, search.id)
        ))
        : fetchingNewsSearches
          ? [getMenuItem(<Spin indicator={<LoadingOutlined />} />, null)]
          : [getMenuItem('No saved searches', null, null, null, null, true)],
      )
    ]),
    [
      saleSearches,
      openKeys,
      fetchingSaleSearches,
      type,
      leaseSearches,
      fetchingLeaseSearches,
      syndicationSearches,
      fetchingSyndicationSearches,
      newsSearches,
      fetchingNewsSearches
    ]
  )
  const rootSubmenuKeys = ['homepage', 'sale', 'lease', 'syndication', 'news', 'normal', 'dseg', 'prism']
  //----------

  /**Refetch data */
  const refetchData = async () => {
    switch (openKeys[0]) {
      case 'sale':
        await fetchSaleSearches()
        return (await fetchSaleSearches()).data
      case 'lease':
        await fetchLeaseSearches()
        return (await fetchLeaseSearches()).data
      case 'syndication':
        await fetchSyndicationSearches()
        return (await fetchSyndicationSearches()).data
      case 'news':
        await fetchNewsSearches()
        return (await fetchNewsSearches()).data
      default:
        break
    }
  }

  /**Handle save searches */
  const { mutateAsync: saveFilters } = useSaveFilters({
    onSuccess: resp => {
      if (resp?.data?.responseData) {
        notification.success({
          message: Messages.SaveSearchSuccess
        })
        refetchData().then(resp => {
          const searches = resp?.data?.responseData
          setSelectedKeys([searches[0].id.toString()])
        })
      } else if (resp?.data?.error) {
        notification.warning({
          message: resp?.data?.error?.message
        })
      }
    },
    onError: (error: ErrorResponse) => {
      notification.warn({
        message: error.message
      })
    }
  })

  const { mutateAsync: editFilters } = useEditFilters({
    onSuccess: resp => {
      if (resp?.data?.responseData) {
        notification.success({
          message: Messages.SaveSearchSuccess
        })
        refetchData()
      } else if (resp?.data?.error) {
        notification.warning({
          message: resp?.data?.error?.message
        })
      }
    },
    onError: (error: ErrorResponse) => {
      notification.warn({
        message: error.message
      })
    }
  })

  const onSaveThisSearch = () => {
    const payload = form.getFieldsValue() as HorseFilters | NewsRequest

    if (!payload.name.replace(/\s/g, '').length) {
      notification.warning({message: 'Please enter title'})
      return
    }

    if (payload.date) {
      payload.date = [dayjs(payload.date.toString()).valueOf()]
    }

    if (payload.price) {
      const prices = payload.price.split('-')
      payload.minPrice = prices[0]
      payload.maxPrice = prices[1]
      delete payload.price
    }

    if (selectedKeys?.length > 0) {
      return editFilters({
        name: payload.name,
        id: parseInt(selectedKeys?.[0]),
        type: type,
        body: payload
      })
    } else {
      return saveFilters({
        name: payload.name,
        type: type,
        body: payload
      })
    }
  }
  //----------

  /**Handle delete search */
  const { mutateAsync: deleteSearch } = useDeleteFilters({
    onSuccess: async resp => {
      if (resp.data.responseData) {
        notification.success({
          message: Messages.DeleteSaveSearchSuccess
        })
        form.resetFields()
        await refetchData().then(resp => {
          const searches = resp?.data?.responseData
          if (searches) {
            setSelectedKeys([searches?.[0]?.id?.toString()])
          } else {
            setSelectedKeys(undefined)
          }
          // switch (openKeys?.[0]) {
          //   case Pages.Sale:
          //     if (saleSearches) {
          //       setSelectedKeys([searches?.[0]?.id?.toString()])
          //     }
          //     break
          //   case Pages.Lease:
          //     if (leaseSearches) {
          //       setSelectedKeys([searches?.[0]?.id?.toString()])
          //     }
          //     break
          //   case Pages.Syndication:
          //     if (syndicationSearches) {
          //       setSelectedKeys([searches?.[0]?.id?.toString()])
          //     }
          //     break
          //   case Pages.News:
          //     if (newsSearches) {
          //       setSelectedKeys([searches?.[0]?.id?.toString()])
          //     }
          //     break
          //   default:
          //     break
          // }
        })
      } else if (resp.data.error) {
        notification.warning({
          message: resp.data.error.message
        })
      }
    },
    onError: error => {
      notification.warning({
        message: error?.message
      })
    },
  })

  const onDelete = (event: React.MouseEvent<HTMLElement, MouseEvent>) => {
    event.stopPropagation()

    Modal.confirm({
      type: 'warning',
      title: 'Confirm',
      content: Messages.ConfirmDeleteSaveSearch,
      okText: 'Yes',
      okType: 'primary',
      cancelText: 'No',
      onOk: async () => {
        form.resetFields()
        return await deleteSearch(parseInt(selectedKeys?.[0]))
      },
    })
  }
  //----------

  /**Handle navigating to corresponding page and search */
  const onSearch = async (event: React.MouseEvent<HTMLElement, MouseEvent>) => {
    event.stopPropagation()

    await onSaveThisSearch()
    await router.push({
      pathname: openKeys[0],
      query: {
        searchId: selectedKeys
      }
    }, openKeys[0])
  }

  return (
    <SearchesTabWrapper>
      <Row
        gutter={16}
      >
        <Col
          // xl={8}
          lg={6}
          md={10}
          span={24}
        >
          <div
            className={
              `xl:max-h-[calc(100vh-272px)] lg:max-h-[calc(100vh-264px)] sticky top-18
              bg-white rounded-sm overflow-y-auto overflow-x-hidden border`
            }
          >
            <Menu
              items={items}
              mode='inline'
              openKeys={openKeys}
              selectedKeys={selectedKeys}
              onOpenChange={onOpenChange}
              onSelect={onSelectedChange}
            />
          </div>
        </Col>
        <Col
          // xl={16}
          lg={18}
          md={14}
          span={24}
        >
          <div
            ref={autoAnimated}
            className='sticky top-36 py-4 bg-white rounded-sm border sm:mt-0 mt-18'
          >
            {openKeys?.length > 0 ? (
              <Spin
                spinning={fetchingSearches}
                indicator={<LoadingOutlined />}
                size='large'
              >
                <SearchesForm
                  page={page}
                  selected={selectedKeys?.[0]}
                  typeId={type}
                  form={form}
                  onSaveThisSearch={onSaveThisSearch}
                  onSearch={onSearch}
                  onDelete={onDelete}
                />
              </Spin>
            ) : (
              <div className='text-base text-center'>
                <p><ExclamationCircleOutlined className='text-secondary-500' /> Please choose a page.</p>
              </div>
            )}
          </div>
        </Col>
      </Row>
    </SearchesTabWrapper>
  )
}

const SearchesTabWrapper = styled.div`
  .ant-collapse {
    > .ant-collapse-item {
      > .ant-collapse-header {
        .ant-collapse-arrow {
          vertical-align: 0.1rem;
        }
      }
    }
  }
`
