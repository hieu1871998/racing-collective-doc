/* eslint-disable @next/next/no-img-element */
/**@prettier */

import { NewsPreview } from '@components'
import styled from '@emotion/styled'
import { News } from '@interfaces'
import { Button, Ellipsis } from '@ui'
import { slugGen } from '@utils'
import { Card, Col, Grid, Row } from 'antd'
import dayjs from 'dayjs'
import parse, { domToReact, Element, HTMLReactParserOptions } from 'html-react-parser'
import Image from 'next/future/image'
import Link from 'next/link'
import React, { useState } from 'react'
import defaultImage from '../../../public/assets/images/logos/thumbnail-primary.svg'


const { useBreakpoint } = Grid

/**
 * MustReadPanel properties.
 * @alias MustReadPanelProps
 * @memberof MustReadPanel
 */
interface MustReadPanelProps {
  /**News data. */
  data: News
  /**Size of the panel. */
  size: 'large' | 'small'
  /**Carousel orientation. */
  orientation?: 'horizontal' | 'vertical'
  /**Active state related to carousel current slide */
  active?: boolean
  /**Handle on click event. */
  onClick?: (event: React.MouseEvent<HTMLDivElement, MouseEvent>) => void
}

/**
 * Panel component for displaying news in Must Read section of News page.
 * @description
 * This component has two variants which is rendered conditionally by passed properties.
 * The large variants is determined by size = 'large' and orientation = 'horizontal',
 * otherwise small variant will be rendered.
 * Large variant is supposed for left side horizontal carousel.
 * Small variant is supposed for right side vertical carousel.
 * @param {MustReadPanelProps} MustReadPanelProps - Component properties.
 * @component
 * @example
 * const data = data
 * const size = 'large'
 * const orientation = 'horizontal'
 * const onClick = () => {}
 *
 * return (
 *  <MustReadPanel
 *    data={data}
 *    size={size}
 *    orientation={orientation}
 *    onClick={onClick}
 *  />
 * )
 * @returns {JSX.Element} MustReadPanel component.
 */
export const MustReadPanel = ({ data, size, orientation, active, onClick }: MustReadPanelProps): JSX.Element => {
  const screens = useBreakpoint()
  const [previewVisible, setPreviewVisible] = useState(false)
  const onAnchorClick = (event: React.MouseEvent<HTMLElement, MouseEvent>) => {
    event.preventDefault()
    setPreviewVisible(true)
  }
  const replaceOptions: HTMLReactParserOptions = {
    replace: ({ name, attribs, children }: Element) => {
      if (name === 'p') {
        return <>{domToReact(children, replaceOptions)}</>
      }
      if (name === 'strong' || (attribs as unknown as string) === 'style') {
        return <>{domToReact(children, replaceOptions)}</>
      }
      if (name === 'iframe' || name === 'img' || name === 'br') {
        return <></>
      }
    },
  }

  return (
    <MustReadPanelWrapper
      className={
        `w-full rounded-sm overflow-hidden
        ${active ? 'bg-white' : 'bg-[#f0f0f3]'}`
      }
      bordered={false}
      onClick={onClick}
    >
      <Row
        className={
          `${size === 'large' ? '' : 'w-full h-[140px] xl:h-[180px] cursor-pointer'}
        `}
      >
        <Col
          span={size === 'large' ? 24 : orientation === 'vertical' ? 10 : 24}
          className={`w-full
          ${size === 'large' ? 'h-52 sm:h-96 lg:h-72 xl:h-96' : 'h-full'}`}
        >
          <Link href={data?.newsUrl ? data.newsUrl : `/news/${data?.id}/${slugGen(data?.title)}`}>
            <a
              target='_blank'
              rel='noreferrer'
            >
              <Image
                className={`
                  w-full h-full cursor-pointer object-cover object-center
                  ${size === 'large' ? 'rounded-t-sm' : 'rounded-l-sm'}
                `}
                // eslint-disable-next-line @typescript-eslint/no-unsafe-assignment
                src={data?.imageUrl || defaultImage}
                fill={!!data?.imageUrl}
                alt={data?.title}
              />
            </a>
          </Link>
        </Col>
        <Col
          className={size === 'large' ? 'bg-white p-4' : 'p-2'}
          span={size === 'large' ? 24 : orientation === 'vertical' ? 14 : 24}
        >
          <Link href={data?.newsUrl ? data.newsUrl : `/news/${data?.id}/${slugGen(data?.title)}`}>
            <a
              className={`w-full font-semibold ${
                size === 'large'
                  ? 'text-xl line-clamp-2 leading-7 min-h-[3em]'
                  : 'xl:text-base text-sm xl:line-clamp-2 line-clamp-1 leading-5 xl:min-h-[48px] xl:mb-0 mb-1'
              }`}
              target='_blank'
              rel='noreferrer'
            >
              <Ellipsis
                title={data?.title}
                maxLines={(screens.xl || size === 'large' || !data?.source?.name) ? 2 : 1}
              >
                {data?.title ?? ''}
              </Ellipsis>
            </a>
          </Link>
          <p className={`font-medium uppercase mb-1 ${size === 'large' ? '' : 'truncate text-xs'}`}>
            {data?.source?.name ?? ''}
          </p>
          <div
            className={`text-xs ${
              size === 'large'
                ? 'line-clamp-3 min-h-[48px]'
                : 'mb-0 line-clamp-2 xl:line-clamp-3 min-h-[2.5em] xl:min-h-[48px]'
            }`}
          >
            {data?.description ? parse(data?.description, replaceOptions) : ''}
          </div>
          <Button
            className={`
              text-xs previewer z-10 px-0
              ${size === 'large' ? '' : 'mb-2'}
            `}
            type='link'
            size='small'
            onClick={onAnchorClick}
          >
            Preview
          </Button>
          {size === 'large' && <div className='border-t w-32' />}
          <p className={`text-[#4F4F4F] ${size === 'large' ? '' : 'text-xs'}`}>
            {dayjs(data?.publishDate).format('DD MMMM YYYY')}
          </p>
        </Col>
      </Row>
      <NewsPreview
        id={data?.id}
        news={data}
        open={previewVisible}
        setVisible={setPreviewVisible}
      />
    </MustReadPanelWrapper>
  )
}

/**
 * Styled Ant Design Card component as wrapper for MustReadPanel component.
 * @alias MustReadPanelWrapper
 * @memberof MustReadPanel
 */
const MustReadPanelWrapper = styled(Card)`
  width: 100%;
  overflow: hidden;
  transition: all 0.3s ease;

  &:hover {
    background-color: #ffffff;

    a {
      color: var(--secondary);
    }

    p {
      a {
        color: inherit;
      }
    }
  }

  a {
    &:hover {
      color: var(--secondary);
    }

    &.previewer {
      color: var(--secondary);

      &::after {
        content: '';
        display: block;
        border-bottom: 1px solid var(--secondary);
        bottom: 4px;
        left: 0;
        right: 0;
        opacity: 0;
        transition: all .3s cubic-bezier(0, 0.5, 0, 1);
      }

      &:hover {
        color: var(--secondary-hover);

        &::after {
          border-color: var(--secondary-hover);
          opacity: 1;
        }
      }
    }
  }

  p {
    a {
      &:hover {
        color: inherit;
      }
    }
  }

  .ant-card-cover {
    img {
      border-radius: 0;
    }
  }

  .ant-card-body {
    padding: 0;
  }
`

export default MustReadPanel
