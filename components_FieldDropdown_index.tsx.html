components/FieldDropdown/index.tsx

import { EllipsisOutlined, LoadingOutlined } from '@ant-design/icons'
import styled from '@emotion/styled'
import { OptionType, SearchFieldProps, SearchSelectField } from '@interfaces'
import { DatePicker, RangePicker, Select, Spin } from '@ui'
import { Col, Form, Grid, Skeleton, Tooltip } from 'antd'
import { DefaultOptionType } from 'antd/lib/select'
import { sortBy } from 'lodash'
import { useEffect, useState } from 'react'

const { useBreakpoint } = Grid

/**
 * Select box component for search fields, already includes Form.Item.
 * @property field
 * @property formInstance
 * @property [index]
 * @property [setFilters]
 * @property [removeFilters]
 * @property [setFieldFilter]
 */
const FieldDropdown = ({
  field,
  formInstance,
  index,
  setFieldFilter,
  filter,
  labelAlign,
  labelCol,
}: SearchFieldProps) => {
  /**Workaround to update page for options sorting */
  const [selectedItems, setSelectedItems] = useState<(string | number)[]>([])
  const [open, setOpen] = useState(false)
  const screens = useBreakpoint()

  /**
   * Get search fields tooltip.
   * @param {SearchSelectField} field
   * @returns {JSX.Element[]} array of selected option labels
   */
  const getFieldTooltip = (field: SearchSelectField): JSX.Element[] => {
    if (field?.mode === 'multiple') {
      const options = field?.options
      const fieldValues: string[] = formInstance.getFieldValue(field.name) as string[]
      let values: React.ReactNode[] = []

      if (field.labelInValue) {
        values = options?.filter(option => fieldValues?.includes(option.value as string)).map(option => option.label)
      } else {
        values = formInstance?.getFieldValue(field.name) as string[]
      }

      const titles = values?.map(value => <p key={value as string}>{value}</p>)

      return titles?.length > 0 ? titles : undefined
    }
  }

  /**
   * Get and sort options by selected options.
   * @param {SearchSelectField} field
   * @returns {OptionType[]} sorted options for select fields
   */
  const getFieldOptions = (field: SearchSelectField): OptionType[] => {
    if (field?.options) {
      if (field?.mode === 'multiple') {
        const fieldValues: (string | number | boolean)[] = formInstance.getFieldValue(field.name) as (string | number)[]
        const options = field?.options

        const sortedOptions = sortBy(options, option => {
          return fieldValues?.indexOf(option.value) !== -1 ? fieldValues?.indexOf(option.value) : options.length
        })

        return sortedOptions
      } else {
        const fieldValue: string | number = formInstance.getFieldValue(field.name) as string | number
        const options = field?.options
        const sortedOptions = sortBy(options, option => {
          return fieldValue === option.value ? 0 : options.length
        })

        return sortedOptions
      }
    } else {
      return [
        {
          label: (
            <Spin indicator={<LoadingOutlined />}>
              <Skeleton
                className='py-2'
                active
                title={false}
                paragraph={{
                  rows: 3,
                }}
              />
            </Spin>
          ),
          value: undefined,
        } as OptionType,
      ]
    }
  }

  const datePicker = (
    <DatePicker
      className='w-full'
      inputReadOnly
      format='DD-MM-YYYY'
      getPopupContainer={trigger => trigger}
      disabledDate={field?.disabledDate}
    />
  )

  const rangePicker = (
    <RangePicker
      className='w-full'
      inputReadOnly
      format='DD-MM-YYYY'
      getPopupContainer={trigger => trigger}
    />
  )


  const select = (
    <Select
      className='w-full'
      mode={field?.mode}
      placeholder={field?.placeholder}
      maxTagCount={'responsive'}
      maxTagTextLength={
        filter
          ? (formInstance.getFieldValue(field?.name) as (string | number)[])?.length > 1 && 5
          : (formInstance.getFieldValue(field?.name) as (string | number)[])?.length > 1 &&
            (screens.xl ? 10 : screens.lg ? 18 : screens.sm ? 10 : 20)
      }
      autoClearSearchValue
      allowClear
      showArrow
      showSearch={field?.showSearch || false}
      filterOption={field?.filterOption || false}
      filterSort={(optionA, optionB) =>
        (optionA.label as unknown as string)
          ?.toLowerCase()
          .localeCompare((optionB.label as unknown as string)?.toLowerCase())
      }
      options={getFieldOptions(field) as DefaultOptionType[]}
      loading={field?.isFetching}
      disabled={field?.disabled}
      onFocus={() => {
        (!!field?.fetch && !field?.options) && field?.fetch()
      }}
      value={selectedItems}
      onChange={(value, options) => {
        field?.onChange && field?.onChange(value, options)
        setSelectedItems(value as number[])
      }}
      onSearch={
        field?.showSearch && !field?.filterOption
          ? value => {
              setFieldFilter({ field, value })
            }
          : undefined
      }
      onBlur={
        filter
          ? () => {
              setFieldFilter({
                field: undefined,
                value: undefined,
              })
            }
          : undefined
      }
      open={open}
      onDropdownVisibleChange={open => setOpen(open)}
      dropdownMatchSelectWidth={false}
      popupClassName='z-50'
    />
  )

  useEffect(() => {
    if (!field?.disabled && field?.fetch && field?.labelInValue) {
      field.fetch()
    }
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [])

  return (
    <>
      {filter ? (
        <Col
          xl={filter ? 3 : undefined}
          lg={filter ? 4 : undefined}
          sm={filter ? 6 : undefined}
          span={filter ? 24 : undefined}
          order={filter ? index + 2 : undefined}
        >
          <FieldSelectWrapper options={field?.options?.length}>
            <Tooltip
              title={getFieldTooltip(field)}
              showArrow={false}
              getPopupContainer={() => document.body}
            >
              <Form.Item
                className='relative'
                name={field?.name}
                label={field?.label}
                labelAlign={labelAlign}
                initialValue={field?.initialValue && field.initialValue}
                noStyle={field?.noStyle}
                rules={field?.rules}
                tooltip={field?.tooltip}
                labelCol={labelCol ? labelCol : undefined}
                colon={false}
              >
                {field?.isDatePicker ? datePicker : field?.isRangePicker ? rangePicker : select}
              </Form.Item>
            </Tooltip>
          </FieldSelectWrapper>
        </Col>
      ) : (
        <FieldSelectWrapper options={field?.options?.length}>
          <Tooltip title={getFieldTooltip(field)}>
            <Form.Item
              className='relative'
              name={field?.name}
              label={field?.label}
              labelAlign={labelAlign}
              initialValue={field?.initialValue && field.initialValue}
              noStyle={field?.noStyle}
              rules={field?.rules}
              tooltip={field?.tooltip}
              labelCol={labelCol ? labelCol : undefined}
              colon={false}
            >
              {field?.isDatePicker ? datePicker : field?.isRangePicker ? rangePicker : select}
            </Form.Item>
          </Tooltip>
        </FieldSelectWrapper>
      )}
    </>
  )
}

export const clearOption: OptionType = {
  label: <EllipsisOutlined />,
  value: undefined,
}

interface FieldSelectWrapperProps {
  options?: number
}

const FieldSelectWrapper = styled.div<FieldSelectWrapperProps>`
  .ant-select-item {
    &.ant-select-item-option {
      &.ant-select-item-option-selected {
        &:not(.ant-select-item-option-disabled) {
          position: sticky;
          z-index: 10;
          top: 0;
        }
      }
    }
  }
`

export { FieldDropdown }
