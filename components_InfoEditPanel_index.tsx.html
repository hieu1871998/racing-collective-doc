components/InfoEditPanel/index.tsx

import { InfoCircleFilled } from '@ant-design/icons'
import { useGetBusinessTypes } from '@api'
import { AboutPanelProps, FieldDropdown } from '@components'
import { Messages } from '@constants'
import { PanelWrapper } from '@containers'
import { useAutoAnimate } from '@formkit/auto-animate/react'
import { OptionType, SearchSelectField } from '@interfaces'
import { Input, Form } from '@ui'
import { Col, Row, Skeleton, Tooltip } from 'antd'
import { useMemo } from 'react'

/**
 * Edit user's personal information panel.
 * @alias InfoEditPanel
 * @author Vi Hieu
 * @category @components
 * @param {AboutPanelProps} props
 * @returns <InfoEditPanel {...props} />
 * @component
 * @example
 * const [form] = Form.useForm()
 * const user = {}
 * const businessTypes = []
 * return (
 *  <InfoEditPanel form={form} user={user} businessTypes={businessTypes} />
 * )
 */
export const InfoEditPanel = ({ form, user }: AboutPanelProps) => {
  const animated = useAutoAnimate(null) as React.Ref<HTMLDivElement>

  const { data: businessTypesResp, refetch, isFetching } = useGetBusinessTypes()
  const businessTypes = useMemo<OptionType[]>(
    () =>
      businessTypesResp?.data?.responseData?.map(type => ({
        value: type?.id,
        label: type?.type,
      })),
    [businessTypesResp]
  )

  const businessTypesField: SearchSelectField = {
    name: 'businessTypeIds',
    label: 'Business type',
    placeholder: 'Business type',
    mode: 'multiple',
    options: businessTypes,
    labelInValue: true,
    fetch: refetch,
    isFetching: isFetching,
    filterOption: (input, option) => (option.children as unknown as string).toLowerCase().includes(input.toLowerCase()),
  }

  const dobField: SearchSelectField = {
    name: 'dob',
    label: 'Date of birth',
    placeholder: 'Date of birth',
    isDatePicker: true,
    initialValue: undefined
  }

  return (
    <PanelWrapper
      className='border-t border-r border-l rounded-t rounded-b-none'
      ref={animated}
      title={<b>Personal Information</b>}
      bordered={false}
    >
      {user ? (
        <Row gutter={12}>
          {user?.roles?.some(role => role?.id === 1) && (
            <Col
              xl={6}
              sm={12}
              span={24}
            >
              <Form.Item
                name='companyName'
                label='Company name'
                tooltip={{
                  title: 'Required field',
                  icon: (
                    <span>
                      <span className='text-red-500'>*</span>
                    </span>
                  ),
                }}
                rules={[
                  {
                    required: true,
                    message: Messages.RequiredField,
                  },
                ]}
              >
                <Input placeholder='Company name' />
              </Form.Item>
            </Col>
          )}
          <Col
            xl={6}
            sm={12}
            span={24}
          >
            <Form.Item
              name='firstName'
              label='First name'
            >
              <Input placeholder='First name' />
            </Form.Item>
          </Col>
          <Col
            xl={6}
            sm={12}
            span={24}
          >
            <Form.Item
              name='lastName'
              label='Last name'
            >
              <Input placeholder='Last name' />
            </Form.Item>
          </Col>
          <Col
            xl={6}
            sm={12}
            span={24}
          >
            <FieldDropdown
              field={dobField}
              formInstance={form}
            />
          </Col>
          <Col
            xl={6}
            sm={12}
            span={24}
          >
            <Form.Item
              name='mobile'
              label='Phone'
              rules={[
                {
                  pattern: /^[0-9]*$/,
                  message: Messages.InvalidPhone,
                },
              ]}
            >
              <Input placeholder='Phone number' />
            </Form.Item>
          </Col>
          <Col
            xl={6}
            sm={12}
            span={24}
          >
            <Form.Item
              name='email'
              label='Email'
            >
              <Input
                placeholder='Email'
                disabled
              />
            </Form.Item>
          </Col>
          <Col
            xl={6}
            sm={12}
            span={24}
          >
            <Form.Item
              name='emailEnquiry'
              label={<p className='truncate'>Email for Sales enquiries</p>}
              tooltip={{
                title: 'Required field',
                icon: (
                  <span>
                    <span className='text-red-500'>*</span>
                  </span>
                ),
              }}
              rules={[
                {
                  required: true,
                  message: Messages.RequiredField,
                },
                {
                  type: 'email',
                  message: Messages.InvalidEmail,
                },
              ]}
            >
              <Input
                placeholder='Email for Sales enquiries'
                suffix={
                  <Tooltip
                    title='This is the email you want all of your sales
                enquiries to go to'
                    color='black'
                  >
                    <span className='cursor-help text-lg text-primary-500'>
                      <InfoCircleFilled />
                    </span>
                  </Tooltip>
                }
              />
            </Form.Item>
          </Col>
          <Col
            xl={6}
            sm={12}
            span={24}
          >
            <FieldDropdown
              field={businessTypesField}
              formInstance={form}
            />
          </Col>
        </Row>
      ) : (
        <Skeleton
          active
          title={{
            width: '100%',
          }}
          paragraph={{
            rows: 4,
          }}
          loading={isFetching}
          round
        />
      )}
    </PanelWrapper>
  )
}

export default InfoEditPanel
