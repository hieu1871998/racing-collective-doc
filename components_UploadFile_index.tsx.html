/**@format */
import { Button, notification, Upload } from '@ui'
import { UploadProps } from 'antd'
import AWS from 'aws-sdk/global'
import S3 from 'aws-sdk/clients/s3'
import { useUploadMedia } from '@api'
import { useContext } from 'react'
import { UserContext } from '@contexts'
import { v4 as uuidv4 } from 'uuid'
import { ErrorResponse, Media, Session } from '@interfaces'
import { map } from 'lodash'
import { PlusOutlined, UploadOutlined } from '@ant-design/icons'
import styled from '@emotion/styled'

/**
 * Get uploaded image URL
 * @alias getUrl
 * @memberof UploadFile
 * @param session
 * @param userId
 * @param guid
 * @param fileName
 * @returns {string} url - URL of the uploaded image
 */
const getUrl = (session: Session, userId: number, guid: string, fileName: string): string => {
  fileName.replace(/[!'()\s]/g, '_')

  const url = `https://${session?.bucket}.s3.${session?.region}.amazonaws.com/bse-upload/image/user/${userId}/${guid}/${fileName}`

  return url
}

export enum MediaTypes {
  Photo = 1,
  Video = 3,
  Audio = 2,
  Album = 4,
  Document = 5,
  Pedigree = 6,
  RaceRecord = 7,
  ApprovalLetter = 8,
}

const getMediaType = (file: File) => {
  if (file.type.includes('video')) return MediaTypes.Video
  if (file.type.includes('audio')) return MediaTypes.Audio
  if (file.type.includes('image')) return MediaTypes.Photo
  return MediaTypes.Document
}

interface UploadFileProps extends UploadProps {
  layout?: 'horizontal' | 'vertical'
  uploadText?: string
  btnType?: 'link' | 'text' | 'ghost' | 'default' | 'primary' | 'dashed'
  direction?: 'ltr' | 'rtl'
}

/**
 * Ant Design Upload component modified for upload various type of files to AWS S3
 * @author Vi Hieu
 * @alias UploadFile
 * @component
 * @param {UploadProps} props
 */
export const UploadFile = (props: UploadFileProps) => {
  const userContext = useContext(UserContext)
  // const [loading, setLoading] = useState(false)
  // const [imageUrl, setImageUrl] = useState<string>()

  const { mutateAsync: getSession } = useUploadMedia({
    onSuccess: resp => {
      if (resp?.data?.responseData) {
        return
      }
    },
    onError: error => {
      console.error(error)
    },
  })

  const { mutateAsync: uploadMedia } = useUploadMedia({
    onSuccess: resp => {
      const logoIdResp = resp?.data?.responseData?.medias?.[0]?.id
      if (logoIdResp) {
        return
      }
    },
    onError: error => {
      console.error(error)
    },
  })

  /**
   * Custom request for handling file upload to AWS S3
   * @param {UploadProps['customRequest']} RcCustomRequestOptions
   */
  const customRequest: UploadProps['customRequest'] = async ({ file, onProgress, onError, onSuccess }) => {
    const sessionResp = await getSession({ newToken: true })
    const session = sessionResp?.data?.responseData?.session
    AWS.config.credentials = new AWS.Credentials(session.accessKeyId, session.secretAccessKey, session.sessionToken)
    AWS.config.update({ region: session.region })

    const s3 = new S3()
    const guid = uuidv4()
    const userId = userContext.currentUser.id
    if (file instanceof File) {
      const fileUrl = getUrl(session, userId, guid, file.name)
      const tagSet = {
        guid: guid,
        creator: userId,
      }
      const dirName = `bse-upload/image/user/${userId}`

      const params: S3.Types.PutObjectRequest = {
        Body: file,
        Bucket: session.bucket,
        Key: `${dirName}/${guid}/${file.name}`,
        ACL: 'public-read',
        ContentType: file.type,
        Tagging: map(tagSet, (tag, key) => `${key}=${tag}`).join('&'),
      }

      const fileData: Media = {
        guid: guid,
        originalFilename: file.name,
        typeOfMedia: getMediaType(file),
        uploader: userId,
        url: fileUrl,
      }

      s3.putObject(params)
        .on('httpUploadProgress', ({ loaded, total }) => {
          onProgress({ percent: (loaded / total) * 100 })
        })
        .send(async (error, data) => {
          if (error) {
            onError(error, fileData)
            console.error(error)
          }
          await uploadMedia({ listMetaData: [fileData] })
            .then(resp => {
              const media = resp?.data?.responseData?.medias[0]
              const error = resp?.data?.error

              if (media) {
                onSuccess(resp?.data?.responseData?.medias[0])
              } else if (error) {
                notification.error({
                  message: 'Upload failed.',
                  description: error.message,
                })
                console.error(error)
              }
            })
            .catch((error: ErrorResponse) => {
              onError(error, fileData)
              console.error(error)
            })
        })

      return fileUrl
    }
  }

  return (
    <UploadFileWrapper>
      <Upload
        listTextBordered
        multiple={props?.multiple}
        customRequest={customRequest}
        layout={props?.layout}
        direction={props?.direction}
        {...props}
      >
        {props.listType === 'picture-card' ? (
          <UploadButtonPictureCard />
        ) : (
          <Button
            type={props?.btnType}
            icon={<UploadOutlined />}
          >
            {props?.uploadText ?? 'Upload'}
          </Button>
        )}
      </Upload>
    </UploadFileWrapper>
  )
}

const UploadButtonPictureCard = () => {
  return (
    <div>
      <PlusOutlined />
      <div className='mt-2'>Upload</div>
    </div>
  )
}

const UploadFileWrapper = styled.div`
  .ant-upload-picture-card-wrapper {
    .ant-upload-list {
      &.ant-upload-list-picture-card {
        .ant-upload {
          &.ant-upload-select {
            &.ant-upload-select-picture-card {
              width: 104px;
              height: 104px;
            }
          }
        }
      }
    }
  }
`

UploadFile.getUrl = getUrl

export default UploadFile
