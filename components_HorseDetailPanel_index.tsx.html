import { CheckOutlined, EditOutlined, LoadingOutlined, ReloadOutlined } from '@ant-design/icons'
import styled from '@emotion/styled'
import { Col, Row, Grid, Divider, Badge, Skeleton, Typography } from 'antd'
import {
  HorseAges,
  ShareOnSocials,
  getHorseLink,
  DetailMediaGallery,
  EnquiryForm,
  HorseEditFormModal,
} from '@components'
import { Horse } from '@interfaces'
import { Button, Spin } from '@ui'
import dayjs from 'dayjs'
import parse, { attributesToProps, domToReact, Element, HTMLReactParserOptions } from 'html-react-parser'
import { HEADER_TYPE, HorseStatusCodes, HorseStatusColorCodes, HorseStatusColors, HorseStatuses } from '@constants/enum'
import { useState } from 'react'
import Image from 'next/future/image'
import defaultThumbnail from '../../../public/assets/images/logos/thumbnail-primary.svg'

const { Title } = Typography

const { useBreakpoint } = Grid

interface HorseDetailPanelProps {
  horse: Horse
  isDetailPage?: boolean
  relist?: () => void
  markAsSold?: () => void
}

/**
 * Horse detail panel, show all information about current horse.
 * @prop {Horse} horse
 * @prop {boolean} isDetailPage
 */
export const HorseDetailPanel = ({ horse, isDetailPage, relist, markAsSold }: HorseDetailPanelProps) => {
  const screens = useBreakpoint()

  const replaceOptions: HTMLReactParserOptions = {
    replace: (domNode: Element) => {
      if (domNode.name === 'iframe') {
        const props = attributesToProps(domNode.attribs)
        return (
          <iframe
            style={{ maxWidth: '100%' }}
            {...props}
          >
            {domToReact(domNode.children)}
          </iframe>
        )
      }
      if (domNode.name === 'a') {
        const props = attributesToProps(domNode.attribs)
        let address: URL = null
        if (domToReact(domNode.children)?.toString().includes('http')) {
          address = new URL(domToReact(domNode.children)?.toString())
        }

        return (
          <p className='truncate'>
            <a
              className='text-primary-500'
              {...props}
            >
              {/* {props.hostname} */}
              {domToReact(domNode.children)?.toString().includes('http')
                ? address?.hostname?.toString()
                : domToReact(domNode.children)}
            </a>
          </p>
        )
      }
    },
  }

  return (
    <PanelWrapper className={`container mx-auto relative bg-white rounded-sm ${isDetailPage ? 'my-4' : 'before:'}`}>
      {!isDetailPage && (
        <ActionMenu
          horse={horse}
          relist={relist}
          markAsSold={markAsSold}
        />
      )}
      <div className='w-full p-4'>
        <Row gutter={24}>
          <Col
            xl={10}
            lg={12}
            span={24}
          >
            <div
              className={`
                  w-full aspect-w-4 aspect-h-3 bg-gray-200
                  ${horse ? '' : 'animate-pulse'}
                `}
            >
              {horse ? (
                <Badge.Ribbon
                  className={`px-4 py-2 h-auto leading-4
                      ${isDetailPage ? 'hidden' : 'uppercase'} 
                      ${horse.status === 1 ? 'animate-pulsing-sm' : ''}
                    `}
                  text={horse?.approved ? HorseStatuses.Published : HorseStatusCodes[horse?.status]}
                  color={horse?.approved ? HorseStatusColors.Green : HorseStatusColorCodes[horse?.status]}
                >
                  <Image
                    id={`${horse?.id}`}
                    className='w-full h-full object-center object-cover'
                    src={
                      horse?.avatar
                        ? horse?.avatar?.cdnMedium ||
                          horse?.avatar?.cdnSmall ||
                          horse?.avatar?.cdnLarge ||
                          horse?.avatar?.cdnOrigin ||
                          (horse?.avatar?.url.includes('http') && horse?.avatar?.url) ||
                          defaultThumbnail
                        : defaultThumbnail
                    }
                    alt=''
                    fill={horse?.avatar !== undefined}
                    sizes='(max-width: 576px) 100vw,
                            (max-width: 1024px) 75vw,
                            50vw'
                  />
                </Badge.Ribbon>
              ) : (
                <Spin
                  className='w-full h-full flex justify-center items-center'
                  size='large'
                  indicator={<LoadingOutlined />}
                />
              )}
            </div>
          </Col>
          <Col
            xl={14}
            lg={12}
            span={24}
          >
            <Skeleton
              className='mt-4'
              loading={!horse}
              active
            >
              <div className='position'>
                <div className='horse-info max-w-[95%] xl:max-w-[100%]'>
                  <Title
                    className='lg:mt-0 mt-4 uppercase'
                    level={screens.xl ? 2 : 4}
                  >
                    {horse?.horseName || horse?.name}
                    {horse?.nickname ? <span> ({horse?.nickname})</span> : <></>}
                  </Title>
                  <p className='xl:text-sm text-xs'>
                    {horse?.age > -1 ? (
                      <strong>
                        <HorseAges age={horse?.age} />{' '}
                      </strong>
                    ) : (
                      <></>
                    )}
                    <b>{horse?.colour && horse?.colour?.colour + ' '}</b>
                    <b>{horse?.sex && horse?.sex + ' '}</b>
                    {(horse?.sire || horse?.dam) && 'by '}
                    <b className='uppercase'>{horse?.sire} </b>
                    {horse?.sire && horse?.dam && 'ex '}
                    <b className='uppercase'>{horse?.dam} </b>
                    {horse?.foaled && `foaled ${dayjs(horse?.foaled).format('DD/MM/YYYY')}`}
                  </p>
                  <div className='mt-4 xl:text-sm text-xs text-neutral-200'>
                    <p>
                      <span>{horse?.countryLocation ? horse?.countryLocation.name : ''}</span>
                      {horse?.state || horse?.location ? ' |' : ''} <span>{horse?.state ? horse?.state.name : ''}</span>{' '}
                      {horse?.location && horse?.state ? '|' : ''} <span>{horse?.location}</span>
                    </p>
                  </div>
                  <p className='mt-4 xl:text-sm text-xs'>
                    {horse?.type?.type} by <b className='text-primary-500'>{horse?.user?.companyName}</b>
                  </p>
                  <p className='mt-4 text-sm'>
                    {horse?.trainerName && horse?.type?.id === HEADER_TYPE.SYNDICATION && (
                      <>
                        To be trained by <b>{horse?.trainerName}</b>
                      </>
                    )}
                  </p>
                  <p className='horse-price mt-2 2xl:text-2xl xl:text-xl text-lg text-primary-500'>
                    <b>
                      {horse?.price
                        ? horse?.price?.toLocaleString('en-AU', {
                            style: 'currency',
                            currency: 'AUD',
                            minimumFractionDigits: 0,
                          })
                        : ''}{' '}
                      <span className='2xl:text-xl xl:text-lg text-base'>
                        {horse?.condition ? horse?.condition : ''}
                      </span>
                    </b>
                  </p>
                  {horse?.bonusScheme && (
                    <div
                      className={`horse-bonuscheme mr-5
                      flex items-center flex-wrap`}
                    >
                      {horse?.bonusScheme.map(bonus =>
                        bonus.media ? (
                          <div
                            key={`${bonus.id}-bonus-key`}
                            className='bonus-item m-2'
                            id={`${bonus.id} BonusKey`}
                          >
                            {/* eslint-disable-next-line @next/next/no-img-element */}
                            <img
                              src={bonus.media.cdnSmall || bonus.media.cdnOrigin}
                              alt={horse?.horseName || horse?.name}
                              title={bonus.code}
                            />
                          </div>
                        ) : (
                          <span
                            key={`${bonus.id}-bonus-key`}
                            className='bonus-item m-2 font-semibold text-white px-2 bg-primary-500'
                          >
                            {bonus.code}
                          </span>
                        )
                      )}
                    </div>
                  )}
                </div>
              </div>
              {isDetailPage && (
                <div
                  className={`share-social-network flex
                  justify-center items-center absolute
                  top-6 sm:right-1 right-0`}
                >
                  <ShareOnSocials
                    data={horse}
                    shareUrl={process.env.NEXT_PUBLIC_ROOT_PATH + getHorseLink(horse).as}
                    type='horse'
                    title={horse?.horseName || horse?.name}
                  />
                </div>
              )}
            </Skeleton>
          </Col>
        </Row>
        <Divider />
        <Row className='w-full'>
          {horse ? (
            <DetailMediaGallery
              isDetailPage={isDetailPage}
              images={horse?.imageMedia}
              videos={horse?.videoMedia}
              audios={horse?.audioMedia}
              documents={horse?.documentMedia}
            />
          ) : (
            <Spin
              className='w-full h-full flex justify-center items-center'
              size='large'
              indicator={<LoadingOutlined />}
            />
          )}
        </Row>
        <Divider />
        <Row>
          <Skeleton
            loading={!horse}
            active
          >
            {horse?.headline && <h5 className='font-bold mb-4'>{horse?.headline}</h5>}
            <div className='w-full text-sm'>{parse(horse?.description ?? '', replaceOptions)}</div>
          </Skeleton>
        </Row>
        {isDetailPage && horse && (
          <>
            <Divider />
            <EnquiryForm
              horseId={horse?.id}
              vendorName={horse?.user?.companyName}
            />
          </>
        )}
      </div>
    </PanelWrapper>
  )
}

        height: 35px;

  p,

  }

  ul {

    li {
      list-style: disc;
      margin-bottom: 1rem;

/**
 * Action menu props.
 */
interface ActionMenuProps {
  /**Horse detail */
  horse: Horse
  /**Relist horse function */
  relist: () => void
  /**Mark horse as sold function */
  markAsSold: () => void
}

/**
 * Action menu for user's available horses detail pane.
 * @component
 */
const ActionMenu = ({ horse, relist, markAsSold }: ActionMenuProps) => {
  const screens = useBreakpoint()
  const [editFormVisible, setEditFormVisible] = useState(false)

  const onEditClick = (event: React.MouseEvent<HTMLElement, MouseEvent>) => {
    event.stopPropagation()
    setEditFormVisible(true)
  }

  const onFormCancel = () => {
    setEditFormVisible(false)
  }

  return (
    <ActionMenuWrapper className='absolute -top-10 right-0 flex flex-row gap-2 z-40'>
      {(horse?.status < 1 || horse?.status > 2 || horse?.approved) && (
        <Button
          type='primary'
          size={screens.xs ? 'small' : 'middle'}
          icon={HorseStatusCodes[horse?.status] === 'sold' ? <ReloadOutlined /> : <CheckOutlined />}
          onClick={event => {
            event.stopPropagation()
            if (HorseStatusCodes[horse?.status] === 'sold') {
              relist()
            } else {
              markAsSold()
            }
          }}
          loading={!horse}
        >
          {horse ? <span>{HorseStatusCodes[horse.status] === 'sold' ? 'RE-LIST' : 'MARK AS SOLD'}</span> : null}
        </Button>
      )}
      <Button
        type='default'
        size={screens.xs ? 'small' : 'middle'}
        icon={<EditOutlined />}
        onClick={onEditClick}
        loading={!horse}
      >
        EDIT
      </Button>
      <HorseEditFormModal
        horseId={horse?.id}
        open={editFormVisible}
        setVisible={setEditFormVisible}
        onCancel={onFormCancel}
      />
    </ActionMenuWrapper>
  )
}

const ActionMenuWrapper = styled.div``

export default HorseDetailPanel
