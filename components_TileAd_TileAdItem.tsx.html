components/TileAd/TileAdItem.tsx

import { ExclamationCircleFilled, MailFilled, PhoneFilled } from '@ant-design/icons'
import styled from '@emotion/styled'
import { Advertisement, Views } from '@interfaces'
import { Tooltip } from 'antd'
import Image from 'next/future/image'
import defaultTileImage from '../../../public/assets/images/background/advdefault.png'
import defaultbannerImage from '../../../public/assets/images/background/banner-bg.png'

interface TileAdItemProps {
  data: Advertisement
  view: Views
}

/**
 * Tile ad item for tile ad carousel component.
 * @alias TileAdItem
 * @author Vi Hieu
 * @component
 * @param {TileAdItemProps} props - TileAdItem component properties.
 * @returns {JSX.Element} TileAdItem component.
 */
export const TileAdItem = ({
  data,
  view
}: TileAdItemProps): JSX.Element => {
  const dataBanner = data?.banner?.url ?? undefined
  const dataBackground = data?.background?.url ?? undefined
  return (
    <TileAdItemWrapper
      className={
        `${view === 'image-list' || view === 'list'
        ? 'relative h-[4.5rem] md:h-32 lg:h-[175px] xl:h-[13.375rem] 2xl:h-64'
        : 'relative xl:h-[465px] lg:h-[510px] sm:h-[545px] h-[495px] rounded'}`
      }
    >
      <a
        href={data?.website}
        target='_blank'
        rel='noreferrer'
      >
        <Image
          className='w-full h-full rounded object-center object-cover'
          src={
            view === 'image-list' || view === 'list'
              ? dataBanner != undefined
                ? dataBanner
                : defaultbannerImage
              : dataBackground != undefined
              ? dataBackground
              : defaultTileImage
          }
          alt=''
          fill
          priority
          sizes='100vw'
        />
      </a>
      {data?.avatar && (
        <a
          href={data?.website}
          target='_blank'
          rel='noreferrer'
        >
          <div className='w-10 h-10 rounded-full absolute top-2 right-2 border shadow z-10'>
            <Image
              className='w-full h-full object-center object-cover rounded-full'
              src={data?.avatar?.cdnMedium ?? data?.avatar?.cdnOrigin ?? data?.avatar?.url}
              alt=''
              fill
            />
          </div>
        </a>
      )}
      <div
        className={
          `w-full absolute bottom-0 left-0 right-0 p-2 text-white rounded-b
          bg-gradient-to-t from-black to-transparent`
        }
      >
        <h3 className='text-white uppercase font-medium text-xl'>{data?.businessName}</h3>
        <div className='flex items-center justify-between'>
          <span>{data?.types[0]?.type}</span>
          <span>{data?.country?.name}</span>
        </div>
        <p>{data?.description}</p>
        <div className='flex justify-between'>
          <a
            className='hover:text-secondary-400'
            href={`tel:${data?.mobile}`}
          >
            <PhoneFilled /> {data?.mobile}
          </a>
          <div className='flex gap-2'>
            <Tooltip
              title='Email'
              color='black'
            >
              <a href={`mailto:${data?.email}`}>
                <MailFilled />
              </a>
            </Tooltip>
            <Tooltip
              title='Visit site'
              color='black'
            >
              <a
                href={data?.website}
                target='_blank'
                rel='noreferrer'
              >
                <ExclamationCircleFilled />
              </a>
            </Tooltip>
          </div>
        </div>
      </div>
    </TileAdItemWrapper>
  )
}

const TileAdItemWrapper = styled.div`
  
`

export default TileAdItem
